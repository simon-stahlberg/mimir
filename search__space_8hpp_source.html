34template<typename... SearchNodeProperties>
35IndexList extract_state_trajectory(const SearchNodeImplVector<SearchNodeProperties...>& search_nodes,
36 ConstSearchNode<SearchNodeProperties...> final_search_node,
37 Index final_state_index)
38{
39 auto trajectory = IndexList {};
40 trajectory.push_back(final_state_index);
41
42 auto cur_search_node = final_search_node;
43
44 while (cur_search_node->get_parent_state() != std::numeric_limits<Index>::max())
45 {
46 trajectory.push_back(cur_search_node->get_parent_state());
47
48 cur_search_node = search_nodes.at(cur_search_node->get_parent_state());
49 }
50
51 std::reverse(trajectory.begin(), trajectory.end());
52
53 return trajectory;
54}
