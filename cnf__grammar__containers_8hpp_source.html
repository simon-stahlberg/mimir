52 template<FeatureCategory D>
53 void push_back(DerivationRule<D> rule)

63 template<FeatureCategory D>
64 const DerivationRuleList<D>& get(NonTerminal<D> non_terminal) const
65 {
66 const auto& container = boost::hana::at_key(m_map, boost::hana::type<D> {});
67
68 auto it = container.find(non_terminal);
69 if (it == container.end())

71 return boost::hana::at_key(empty_lists, boost::hana::type<D> {});

74 return it->second;
