152inline std::ostream& operator<<(std::ostream& os, const Statistics& statistics)
153{
154 os << "[BrFS] Search time: " << statistics.get_search_time_ms().count() << "ms" << "\n"
155 << "[BrFS] Number of generated states: " << statistics.get_num_generated() << "\n"
156 << "[BrFS] Number of expanded states: " << statistics.get_num_expanded() << "\n"
157 << "[BrFS] Number of pruned states: " << statistics.get_num_pruned() << "\n"
158 << "[BrFS] Number of generated states until last g-layer: "
159 << (statistics.get_num_generated_until_g_value().empty() ? 0 : statistics.get_num_generated_until_g_value().back()) << "\n"
160 << "[BrFS] Number of expanded states until last g-layer: "
161 << (statistics.get_num_expanded_until_g_value().empty() ? 0 : statistics.get_num_expanded_until_g_value().back()) << "\n"
162 << "[BrFS] Number of pruned states until last g-layer: "
163 << (statistics.get_num_pruned_until_g_value().empty() ? 0 : statistics.get_num_pruned_until_g_value().back()) << "\n"
164 << "[BrFS] Number of reached fluent atoms: " << statistics.get_num_reached_fluent_atoms() << "\n"
165 << "[BrFS] Number of reached derived atoms: " << statistics.get_num_reached_derived_atoms() << "\n"
166 << "[BrFS] Number of nodes: " << statistics.get_num_nodes() << "\n"
167 << "[BrFS] Number of bytes for nodes: " << statistics.get_num_bytes_for_nodes() << "\n"
168 << "[BrFS] Number of bytes per node: " << divide_or_zero(statistics.get_num_bytes_for_nodes(), statistics.get_num_nodes()) << "\n"
169 << "[BrFS] Number of bytes for problem: " << statistics.get_num_bytes_for_problem() << "\n"
170 << "[BrFS] Total number of bytes used: " << (statistics.get_num_bytes_for_problem() + statistics.get_num_bytes_for_nodes());
171
172 return os;
173}
