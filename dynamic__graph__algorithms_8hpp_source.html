47template<typename Graph, IsDirection Direction>
48 requires IsDynamicGraph<std::decay_t<Graph>> && IsVertexListGraph<Graph> && IsIncidenceGraph<Graph>
49auto strong_components(const DirectionTaggedType<Graph, Direction>& g)
50{
51 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
52 using vertices_size_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type;
53 using ComponentMap = UnorderedMapReadWritePropertyMap<vertex_descriptor_type, vertices_size_type>;
54 using VertexIndexMap = UnorderedMapReadPropertyMap<vertex_descriptor_type, vertex_descriptor_type>;
55
56 auto vertex_remap = std::unordered_map<vertex_descriptor_type, vertex_descriptor_type>();
57 for (vertex_descriptor_type v : g.get().get_vertex_indices())
58 {
59 vertex_remap[v] = vertex_remap.size();
60 }
61 auto vertex_index_map = VertexIndexMap(vertex_remap);
62
63 auto c = std::unordered_map<vertex_descriptor_type, vertices_size_type>();
64 auto component_map = ComponentMap(c);
65
66 const auto num_components = boost::strong_components(g, component_map, boost::vertex_index_map(vertex_index_map));
67 return std::make_pair(num_components, c);
68}

74template<typename Graph, IsDirection Direction, class SourceInputIter>
75 requires IsDynamicGraph<std::decay_t<Graph>> && IsVertexListGraph<Graph> && IsIncidenceGraph<Graph>
76auto breadth_first_search(const DirectionTaggedType<Graph, Direction>& g, SourceInputIter s_begin, SourceInputIter s_end)
77{
78 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
79 using edge_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::edge_descriptor;
80 using PredecessorMap = UnorderedMapReadWritePropertyMap<vertex_descriptor_type, vertex_descriptor_type>;
81 using DistanceMap = UnorderedMapReadWritePropertyMap<vertex_descriptor_type, DiscreteCost>;
82 using ColorMap = UnorderedMapReadWritePropertyMap<vertex_descriptor_type, boost::default_color_type>;
83
84 auto buffer = boost::queue<vertex_descriptor_type>();
85 auto p = std::unordered_map<vertex_descriptor_type, vertex_descriptor_type>();
86 for (vertex_descriptor_type v : g.get().get_vertex_indices())
87 {
88 p[v] = v;
89 }
90 auto inf = std::numeric_limits<DiscreteCost>::max();
91 auto d = std::unordered_map<vertex_descriptor_type, DiscreteCost>();
92 for (vertex_descriptor_type v : g.get().get_vertex_indices())
93 {
94 d[v] = inf;
95 }
96 for (auto it = s_begin; it != s_end; ++it)
97 {
98 d[*it] = DiscreteCost(0);
99 }

101 struct BFSBoostVisitor : public boost::bfs_visitor<>
102 {
103 PredecessorMap predecessors;
104 DistanceMap distances;
105
106 BFSBoostVisitor(PredecessorMap predecessors, DistanceMap distances) : predecessors(predecessors), distances(distances) {}
107
108 void tree_edge(edge_descriptor_type edge, const DirectionTaggedType<Graph, Direction>& graph)
109 {
110 auto u = mimir::graphs::source(edge, graph);
111 auto v = mimir::graphs::target(edge, graph);
112 predecessors.set(v, u);
113 distances.set(v, distances.get()[u] + 1);
114 }
115 };
116 auto visitor = BFSBoostVisitor(PredecessorMap(p), DistanceMap(d));
117
118 auto c = std::unordered_map<vertex_descriptor_type, boost::default_color_type>();
119 auto color_map = ColorMap(c);
120 boost::breadth_first_search(g, s_begin, s_end, buffer, visitor, color_map);
121
122 return std::make_tuple(p, d);
123}

198template<typename Graph, IsDirection Direction, class SourceInputIter>
199 requires IsDynamicGraph<std::decay_t<Graph>> && IsVertexListGraph<std::decay_t<Graph>> && IsIncidenceGraph<std::decay_t<Graph>>
200auto dijkstra_shortest_paths(const DirectionTaggedType<Graph, Direction>& g, const ContinuousCostMap& w, SourceInputIter s_begin, SourceInputIter s_end)
201{
202 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
203 using edge_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::edge_descriptor;
204 using VertexIndexMap = UnorderedMapReadPropertyMap<vertex_descriptor_type, vertex_descriptor_type>;
205
206 auto p = std::unordered_map<vertex_descriptor_type, vertex_descriptor_type>();
207 auto predecessor_map = UnorderedMapReadWritePropertyMap<vertex_descriptor_type, vertex_descriptor_type>(p);
208 auto d = ContinuousCostMap();
209 auto distance_map = UnorderedMapReadWritePropertyMap<vertex_descriptor_type, ContinuousCost>(d);
210 auto weight_map = UnorderedMapReadPropertyMap<edge_descriptor_type, ContinuousCost>(w);
211 auto vertex_remap = std::unordered_map<vertex_descriptor_type, vertex_descriptor_type>();
212 for (vertex_descriptor_type v : g.get().get_vertex_indices())
213 {
214 vertex_remap[v] = vertex_remap.size();
215 }
216 auto vertex_index_map = VertexIndexMap(vertex_remap);
217 auto compare = std::less<ContinuousCost>();
218 auto combine = std::plus<ContinuousCost>();
219 auto inf = std::numeric_limits<ContinuousCost>::infinity();
220 auto zero = ContinuousCost();
221
222 // multiple source shortest path version.
223 boost::dijkstra_shortest_paths(g, //
224 s_begin,
225 s_end,
226 predecessor_map,
227 distance_map,
228 weight_map,
229 vertex_index_map,
230 compare,
231 combine,
232 inf,
233 zero,
234 boost::default_dijkstra_visitor());
235
236 return std::make_tuple(p, d);
237};
