102template<typename Derived_>
103class StaticAStarAlgorithmEventHandlerBase : public IAStarAlgorithmEventHandler
104{
105protected:
106 AStarAlgorithmStatistics m_statistics;
107 formalism::Problem m_problem;
108 bool m_quiet;
109
110private:
111 StaticAStarAlgorithmEventHandlerBase() = default;
112 friend Derived_;
113
115 constexpr const auto& self() const { return static_cast<const Derived_&>(*this); }
116 constexpr auto& self() { return static_cast<Derived_&>(*this); }
117
118public:
119 StaticAStarAlgorithmEventHandlerBase(formalism::Problem problem, bool quiet = true) : m_statistics(), m_problem(problem), m_quiet(quiet) {}
120
121 void on_expand_state(State state) override
122 {
123 m_statistics.increment_num_expanded();

125 if (!m_quiet)

127 self().on_expand_state_impl(state);
128 }
129 }
130
131 void on_expand_goal_state(State state) override
132 {
133 if (!m_quiet)

135 self().on_expand_goal_state_impl(state);
136 }
137 }
138
139 void on_generate_state(State state, formalism::GroundAction action, ContinuousCost action_cost, State successor_state) override
140 {
141 m_statistics.increment_num_generated();
142
143 if (!m_quiet)

145 self().on_generate_state_impl(state, action, action_cost, successor_state);
146 }
147 }
148
149 void on_generate_state_relaxed(State state, formalism::GroundAction action, ContinuousCost action_cost, State successor_state) override
150 {
151 if (!m_quiet)

153 self().on_generate_state_relaxed_impl(state, action, action_cost, successor_state);
154 }
155 }
156
157 void on_generate_state_not_relaxed(State state, formalism::GroundAction action, ContinuousCost action_cost, State successor_state) override
158 {
159 if (!m_quiet)
160 {
161 self().on_generate_state_relaxed_impl(state, action, action_cost, successor_state);
162 }
163 }
164
165 void on_close_state(State state) override
166 {
167 if (!m_quiet)
168 {
169 self().on_close_state_impl(state);
170 }
171 }
172
173 void on_finish_f_layer(double f_value) override
174 {
175 m_statistics.on_finish_f_layer(f_value);
176
177 if (!m_quiet)
178 {
179 assert(!m_statistics.get_num_expanded_until_f_value().empty());
180 self().on_finish_f_layer_impl(f_value,
181 m_statistics.get_num_expanded_until_f_value().rbegin()->second,
182 m_statistics.get_num_generated_until_f_value().rbegin()->second);
183 }
184 }
185
186 void on_prune_state(State state) override
187 {
188 m_statistics.increment_num_pruned();
189
190 if (!m_quiet)
191 {
192 self().on_prune_state_impl(state);
193 }
194 }
195
196 void on_start_search(State start_state) override
197 {
198 m_statistics = AStarAlgorithmStatistics();
199
200 m_statistics.set_search_start_time_point(std::chrono::high_resolution_clock::now());
201
202 if (!m_quiet)
203 {
204 self().on_start_search_impl(start_state);
205 }
206 }
207
208 void on_end_search(uint64_t num_reached_fluent_atoms,
209 uint64_t num_reached_derived_atoms,
210 uint64_t num_bytes_for_unextended_state_portion,
211 uint64_t num_bytes_for_extended_state_portion,
212 uint64_t num_bytes_for_nodes,
213 uint64_t num_bytes_for_actions,
214 uint64_t num_bytes_for_axioms,
215 uint64_t num_states,
216 uint64_t num_nodes,
217 uint64_t num_actions,
218 uint64_t num_axioms) override
219
220 {
221 m_statistics.set_search_end_time_point(std::chrono::high_resolution_clock::now());
222 m_statistics.set_num_reached_fluent_atoms(num_reached_fluent_atoms);
223 m_statistics.set_num_reached_derived_atoms(num_reached_derived_atoms);
224 m_statistics.set_num_bytes_for_unextended_state_portion(num_bytes_for_unextended_state_portion);
225 m_statistics.set_num_bytes_for_extended_state_portion(num_bytes_for_extended_state_portion);
226 m_statistics.set_num_bytes_for_nodes(num_bytes_for_nodes);
227 m_statistics.set_num_bytes_for_actions(num_bytes_for_actions);
228 m_statistics.set_num_bytes_for_axioms(num_bytes_for_axioms);
229 m_statistics.set_num_states(num_states);
230 m_statistics.set_num_nodes(num_nodes);
231 m_statistics.set_num_actions(num_actions);
232 m_statistics.set_num_axioms(num_axioms);
233
234 if (!m_quiet)
235 {
236 self().on_end_search_impl(num_reached_fluent_atoms,
237 num_reached_derived_atoms,
238 num_bytes_for_unextended_state_portion,
239 num_bytes_for_extended_state_portion,
240 num_bytes_for_nodes,
241 num_bytes_for_actions,
242 num_bytes_for_axioms,
243 num_states,
244 num_nodes,
245 num_actions,
246 num_axioms);
247 }
248 }
249
250 void on_solved(const Plan& plan) override
251 {
252 if (!m_quiet)
253 {
254 self().on_solved_impl(plan);
255 }
256 }
257
258 void on_unsolvable() override
259 {
260 if (!m_quiet)
261 {
262 self().on_unsolvable_impl();
263 }
264 }
265
266 void on_exhausted() override
267 {
268 if (!m_quiet)
269 {
270 self().on_exhausted_impl();
271 }
272 }
273
278 const AStarAlgorithmStatistics& get_statistics() const override { return m_statistics; }
279 bool is_quiet() const { return m_quiet; }
280};
