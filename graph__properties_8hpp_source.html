32template<typename G>
33 requires IsEdgeListGraph<G> //
34bool is_undirected(const G& graph)
35{
36 // Create datastructure for efficient lookup
37 auto directed_edges = UnorderedSet<IndexPair>();
38 for (const auto& e_idx : graph.get_edge_indices())
39 {
40 const auto& edge = graph.get_edge(e_idx);
41 directed_edges.emplace(edge.get_source(), edge.get_target());
42 }
43
44 for (const auto& e_idx : graph.get_edge_indices())
45 {
46 const auto& edge = graph.get_edge(e_idx);
47 if (!directed_edges.contains(std::make_pair(edge.get_target(), edge.get_source())))
48 {
49 return false; // found no matching anti-parallel edge => directed graph
50 }
51 }
52
53 return true; // found matching parallel edges => undirected graph
54}

56template<typename G>
57 requires IsEdgeListGraph<G> //
58bool is_multi(const G& graph)
59{
60 auto directed_edges = UnorderedSet<IndexPair>();
61 for (const auto& e_idx : graph.get_edge_indices())
62 {
63 const auto& edge = graph.get_edge(e_idx);
64 if (!directed_edges.emplace(edge.get_source(), edge.get_target()).second)
65 {
66 return true; // found parallel edge => multi-graph
67 }
68 }
69 return false; // found no parallel edges => not multi-graph
70}

78 const auto& edge = graph.get_edge(e_idx);
