92template<typename Derived_>
93class BrFSAlgorithmEventHandlerBase : public IBrFSAlgorithmEventHandler
94{
95protected:
96 BrFSAlgorithmStatistics m_statistics;
97 Problem m_problem;
98 bool m_quiet;
99
100private:
101 BrFSAlgorithmEventHandlerBase() = default;
102 friend Derived_;
103
105 constexpr const auto& self() const { return static_cast<const Derived_&>(*this); }
106 constexpr auto& self() { return static_cast<Derived_&>(*this); }
107
108public:
109 explicit BrFSAlgorithmEventHandlerBase(Problem problem, bool quiet = true) : m_statistics(), m_problem(problem), m_quiet(quiet) {}
110
111 void on_expand_state(State state) override
112 {
113 m_statistics.increment_num_expanded();
114
115 if (!m_quiet)
116 {
117 self().on_expand_state_impl(state);
118 }
119 }
120
121 void on_expand_goal_state(State state) override
122 {
123 if (!m_quiet)
124 {
125 self().on_expand_goal_state_impl(state);
126 }
127 }
128
129 void on_generate_state(State state, GroundAction action, ContinuousCost action_cost, State successor_state) override
130 {
131 m_statistics.increment_num_generated();
132
133 if (!m_quiet)
134 {
135 self().on_generate_state_impl(state, action, action_cost, successor_state);
136 }
137 }
138
139 void on_generate_state_in_search_tree(State state, GroundAction action, ContinuousCost action_cost, State successor_state) override
140 {
141 if (!m_quiet)
142 {
143 self().on_generate_state_in_search_tree_impl(state, action, action_cost, successor_state);
144 }
145 }
146
147 void on_generate_state_not_in_search_tree(State state, GroundAction action, ContinuousCost action_cost, State successor_state) override
148 {
149 if (!m_quiet)
150 {
151 self().on_generate_state_not_in_search_tree_impl(state, action, action_cost, successor_state);
152 }
153 }
154
155 void on_finish_g_layer() override
156 {
157 m_statistics.on_finish_g_layer();
158
159 if (!m_quiet)
160 {
161 assert(!m_statistics.get_num_expanded_until_g_value().empty());
162 self().on_finish_g_layer_impl(m_statistics.get_num_expanded_until_g_value().size() - 1,
163 m_statistics.get_num_expanded_until_g_value().back(),
164 m_statistics.get_num_generated_until_g_value().back());
165 }
166 }
167
168 void on_start_search(State start_state) override
169 {
170 m_statistics = BrFSAlgorithmStatistics();
171
172 m_statistics.set_search_start_time_point(std::chrono::high_resolution_clock::now());
173
174 if (!m_quiet)
175 {
176 self().on_start_search_impl(start_state);
177 }
178 }
179
180 void on_end_search(uint64_t num_reached_fluent_atoms,
181 uint64_t num_reached_derived_atoms,
182 uint64_t num_bytes_for_unextended_state_portion,
183 uint64_t num_bytes_for_extended_state_portion,
184 uint64_t num_bytes_for_nodes,
185 uint64_t num_bytes_for_actions,
186 uint64_t num_bytes_for_axioms,
187 uint64_t num_states,
188 uint64_t num_nodes,
189 uint64_t num_actions,
190 uint64_t num_axioms) override
191 {
192 m_statistics.set_search_end_time_point(std::chrono::high_resolution_clock::now());
193 m_statistics.set_num_reached_fluent_atoms(num_reached_fluent_atoms);
194 m_statistics.set_num_reached_derived_atoms(num_reached_derived_atoms);
195 m_statistics.set_num_bytes_for_unextended_state_portion(num_bytes_for_unextended_state_portion);
196 m_statistics.set_num_bytes_for_extended_state_portion(num_bytes_for_extended_state_portion);
197 m_statistics.set_num_bytes_for_nodes(num_bytes_for_nodes);
198 m_statistics.set_num_bytes_for_actions(num_bytes_for_actions);
199 m_statistics.set_num_bytes_for_axioms(num_bytes_for_axioms);
200 m_statistics.set_num_states(num_states);
201 m_statistics.set_num_nodes(num_nodes);
202 m_statistics.set_num_actions(num_actions);
203 m_statistics.set_num_axioms(num_axioms);
204
205 if (!m_quiet)
206 {
207 self().on_end_search_impl(num_reached_fluent_atoms,
208 num_reached_derived_atoms,
209 num_bytes_for_unextended_state_portion,
210 num_bytes_for_extended_state_portion,
211 num_bytes_for_nodes,
212 num_bytes_for_actions,
213 num_bytes_for_axioms,
214 num_states,
215 num_nodes,
216 num_actions,
217 num_axioms);
218 }
219 }
220
221 void on_solved(const Plan& plan) override
222 {
223 if (!m_quiet)
224 {
225 self().on_solved_impl(plan);
226 }
227 }
228
229 void on_unsolvable() override
230 {
231 if (!m_quiet)
232 {
233 self().on_unsolvable_impl();
234 }
235 }
236
237 void on_exhausted() override
238 {
239 if (!m_quiet)
240 {
241 self().on_exhausted_impl();
242 }
243 }
244
246 const BrFSAlgorithmStatistics& get_statistics() const override { return m_statistics; }
247 bool is_quiet() const override { return m_quiet; }
248};
