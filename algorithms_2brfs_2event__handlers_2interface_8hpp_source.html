91template<typename Derived_>
92class EventHandlerBase : public IEventHandler
93{
94protected:
95 Statistics m_statistics;
96 formalism::Problem m_problem;
97 bool m_quiet;
98
99private:
100 EventHandlerBase() = default;
101 friend Derived_;
102
104 constexpr const auto& self() const { return static_cast<const Derived_&>(*this); }
105 constexpr auto& self() { return static_cast<Derived_&>(*this); }
106
107public:
108 explicit EventHandlerBase(formalism::Problem problem, bool quiet = true) : m_statistics(), m_problem(problem), m_quiet(quiet) {}
109
110 void on_expand_state(State state) override
111 {
112 m_statistics.increment_num_expanded();
113
114 if (!m_quiet)
115 {
116 self().on_expand_state_impl(state);
117 }
118 }
119
120 void on_expand_goal_state(State state) override
121 {
122 if (!m_quiet)
123 {
124 self().on_expand_goal_state_impl(state);
125 }
126 }
127
128 void on_generate_state(State state, formalism::GroundAction action, ContinuousCost action_cost, State successor_state) override
129 {
130 m_statistics.increment_num_generated();
131
132 if (!m_quiet)
133 {
134 self().on_generate_state_impl(state, action, action_cost, successor_state);
135 }
136 }
137
138 void on_generate_state_in_search_tree(State state, formalism::GroundAction action, ContinuousCost action_cost, State successor_state) override
139 {
140 if (!m_quiet)
141 {
142 self().on_generate_state_in_search_tree_impl(state, action, action_cost, successor_state);
143 }
144 }
145
146 void on_generate_state_not_in_search_tree(State state, formalism::GroundAction action, ContinuousCost action_cost, State successor_state) override
147 {
148 if (!m_quiet)
149 {
150 self().on_generate_state_not_in_search_tree_impl(state, action, action_cost, successor_state);
151 }
152 }
153
154 void on_finish_g_layer(DiscreteCost g_value) override
155 {
156 m_statistics.on_finish_g_layer();
157
158 if (!m_quiet)
159 {
160 assert(!m_statistics.get_num_expanded_until_g_value().empty());
161 self().on_finish_g_layer_impl(g_value, m_statistics.get_num_expanded_until_g_value().back(), m_statistics.get_num_generated_until_g_value().back());
162 }
163 }
164
165 void on_start_search(State start_state) override
166 {
167 m_statistics = Statistics();
168
169 m_statistics.set_search_start_time_point(std::chrono::high_resolution_clock::now());
170
171 if (!m_quiet)
172 {
173 self().on_start_search_impl(start_state);
174 }
175 }
176
177 void on_end_search(uint64_t num_reached_fluent_atoms,
178 uint64_t num_reached_derived_atoms,
179 uint64_t num_bytes_for_unextended_state_portion,
180 uint64_t num_bytes_for_extended_state_portion,
181 uint64_t num_bytes_for_nodes,
182 uint64_t num_bytes_for_actions,
183 uint64_t num_bytes_for_axioms,
184 uint64_t num_states,
185 uint64_t num_nodes,
186 uint64_t num_actions,
187 uint64_t num_axioms) override
188 {
189 m_statistics.set_search_end_time_point(std::chrono::high_resolution_clock::now());
190 m_statistics.set_num_reached_fluent_atoms(num_reached_fluent_atoms);
191 m_statistics.set_num_reached_derived_atoms(num_reached_derived_atoms);
192 m_statistics.set_num_bytes_for_unextended_state_portion(num_bytes_for_unextended_state_portion);
193 m_statistics.set_num_bytes_for_extended_state_portion(num_bytes_for_extended_state_portion);
194 m_statistics.set_num_bytes_for_nodes(num_bytes_for_nodes);
195 m_statistics.set_num_bytes_for_actions(num_bytes_for_actions);
196 m_statistics.set_num_bytes_for_axioms(num_bytes_for_axioms);
197 m_statistics.set_num_states(num_states);
198 m_statistics.set_num_nodes(num_nodes);
199 m_statistics.set_num_actions(num_actions);
200 m_statistics.set_num_axioms(num_axioms);
201
202 if (!m_quiet)
203 {
204 self().on_end_search_impl(num_reached_fluent_atoms,
205 num_reached_derived_atoms,
206 num_bytes_for_unextended_state_portion,
207 num_bytes_for_extended_state_portion,
208 num_bytes_for_nodes,
209 num_bytes_for_actions,
210 num_bytes_for_axioms,
211 num_states,
212 num_nodes,
213 num_actions,
214 num_axioms);
215 }
216 }
217
218 void on_solved(const Plan& plan) override
219 {
220 if (!m_quiet)
221 {
222 self().on_solved_impl(plan);
223 }
224 }
225
226 void on_unsolvable() override
227 {
228 if (!m_quiet)
229 {
230 self().on_unsolvable_impl();
231 }
232 }
233
234 void on_exhausted() override
235 {
236 if (!m_quiet)
237 {
238 self().on_exhausted_impl();
239 }
240 }
241
243 const Statistics& get_statistics() const override { return m_statistics; }
244 bool is_quiet() const { return m_quiet; }
245};
