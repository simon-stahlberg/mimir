129 auto consistent_vertices(const StaticAssignmentSets& static_details, const DynamicAssignmentSets& dynamic_details) const
130 {
131 return get_vertices()
132 | std::views::filter(
133 [this, &static_details, &dynamic_details](auto&& vertex)
134 {
135 return vertex.consistent_literals(m_condition->get_literals<formalism::FluentTag>(), dynamic_details.fluent_predicate_assignment_sets)
136 && vertex.consistent_literals(m_condition->get_literals<formalism::DerivedTag>(),
137 dynamic_details.derived_predicate_assignment_sets)
138 && vertex.consistent_literals(m_condition->get_numeric_constraints(),
139 static_details.static_function_skeleton_assignment_sets,

141 });

148 auto consistent_edges(const StaticAssignmentSets& static_details, const DynamicAssignmentSets& dynamic_details) const
149 {
150 static thread_local SharedObjectPool<boost::dynamic_bitset<>> s_pool;
151 auto vertex_mask = s_pool.get_or_allocate();
152 vertex_mask->resize(get_num_vertices());
153 vertex_mask->reset();
154 for (const auto& v : consistent_vertices(static_details, dynamic_details))
155 vertex_mask->set(v.get_index());
156
157 return get_edges()
158 | std::views::filter(
159 [this, mask = std::move(vertex_mask), &static_details, &dynamic_details](auto&& edge)
160 {
161 return mask->test(edge.get_src().get_index()) && mask->test(edge.get_dst().get_index())
162 && edge.consistent_literals(m_condition->get_literals<formalism::FluentTag>(), dynamic_details.fluent_predicate_assignment_sets)
163 && edge.consistent_literals(m_condition->get_literals<formalism::DerivedTag>(), dynamic_details.derived_predicate_assignment_sets)
164 && edge.consistent_literals(m_condition->get_numeric_constraints(),
165 static_details.static_function_skeleton_assignment_sets,
166 dynamic_details.fluent_function_skeleton_assignment_sets);
167 });
168 }

190 void advance()
191 {
192 if (++m_targets_pos >= get_graph().m_target_offsets[m_sources_pos])
193 ++m_sources_pos;
194 }
195
196 public:
197 using difference_type = std::ptrdiff_t;
198 using value_type = Edge;
199 using pointer = value_type*;
200 using reference = const value_type&;
201 using iterator_category = std::forward_iterator_tag;
202
203 EdgeIterator() : m_graph(nullptr), m_sources_pos(0), m_targets_pos(0) {}
204 EdgeIterator(const StaticConsistencyGraph& graph, bool begin) :
205 m_graph(&graph),
206 m_sources_pos(begin ? 0 : graph.m_sources.size()),
207 m_targets_pos(begin ? 0 : graph.m_targets.size())
208 {
209 }
210 value_type operator*() const
211 {
212 assert(m_sources_pos < get_graph().m_sources.size());
213 assert(m_targets_pos < get_graph().m_targets.size());
214 return Edge(get_graph().m_vertices[get_graph().m_sources[m_sources_pos]], get_graph().m_vertices[get_graph().m_targets[m_targets_pos]]);
215 }
216 EdgeIterator& operator++()
217 {
218 advance();
219 return *this;
220 }
221 EdgeIterator operator++(int)
222 {
223 EdgeIterator tmp = *this;
224 ++(*this);
225 return tmp;
226 }
227 bool operator==(const EdgeIterator& other) const { return m_targets_pos == other.m_targets_pos && m_sources_pos == other.m_sources_pos; }
228 bool operator!=(const EdgeIterator& other) const { return !(*this == other); }
229 };
