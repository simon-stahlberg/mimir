164template<typename _Alloc>
165class __promise_base_alloc
166{
167 static constexpr std::size_t __offset_of_allocator(std::size_t __frameSize) noexcept { return __aligned_allocation_size(__frameSize, alignof(_Alloc)); }
168
169 static constexpr std::size_t __padded_frame_size(std::size_t __frameSize) noexcept { return __offset_of_allocator(__frameSize) + sizeof(_Alloc); }
170
171 static _Alloc& __get_allocator(void* __frame, std::size_t __frameSize) noexcept
172 {
173 return *reinterpret_cast<_Alloc*>(static_cast<char*>(__frame) + __offset_of_allocator(__frameSize));
174 }
175
176public:
177 template<typename... _Args>
178 static void* operator new(std::size_t __frameSize, std::allocator_arg_t, _Alloc __alloc, _Args&...)
179 {
180 void* __frame = __alloc.allocate(__padded_frame_size(__frameSize));
181
182 // Store allocator at end of the coroutine frame.
183 // Assuming the allocator's move constructor is non-throwing (a requirement for allocators)
184 ::new (static_cast<void*>(std::addressof(__get_allocator(__frame, __frameSize)))) _Alloc(std::move(__alloc));
185
186 return __frame;
187 }
188
189 template<typename _This, typename... _Args>
190 static void* operator new(std::size_t __frameSize, _This&, std::allocator_arg_t, _Alloc __alloc, _Args&...)
191 {
192 return __promise_base_alloc::operator new(__frameSize, std::allocator_arg, std::move(__alloc));
193 }
194
195 static void operator delete(void* __ptr, std::size_t __frameSize) noexcept
196 {
197 _Alloc& __alloc = __get_allocator(__ptr, __frameSize);
198 _Alloc __localAlloc(std::move(__alloc));
199 __alloc.~Alloc();
200 __localAlloc.deallocate(static_cast<std::byte*>(__ptr), __padded_frame_size(__frameSize));
201 }

320 // set the parent, root and exceptions pointer and
321 // resume the nested
322 template<typename _Promise>
323 std::coroutine_handle<> await_suspend(std::coroutine_handle<_Promise> __h) noexcept
324 {
325 __generator_promise_base& __current = __h.promise();
326 __generator_promise_base& __nested = *__gen_.__get_promise();
327 __generator_promise_base& __root = *__current.__root_;
328
329 __nested.__root_ = __current.__root_;
330 __nested.__parentOrLeaf_ = __h;
331
332 // Lazily construct the __exception_ member here now that we
333 // know it will be used as a nested generator. This will be
334 // destroyed by the promise destructor.
335 __nested.__exception_.construct();
336 __root.__parentOrLeaf_ = __gen_.__get_coro();
337
338 // Immediately resume the nested coroutine (nested generator)
339 return __gen_.__get_coro();
340 }

342 void await_resume()
343 {
344 __generator_promise_base& __nestedPromise = *__gen_.__get_promise();
345 if (__nestedPromise.__exception_.get())
346 {
347 std::rethrow_exception(std::move(__nestedPromise.__exception_.get()));
348 }
349 }
