115template<IsStaticOrFluentOrDerivedTag... Ps>
116size_t GroundConjunctiveConditionImpl::get_num_preconditions() const
117{
118 auto result = size_t(0);
119 boost::hana::for_each(get_hana_compressed_precondition<Ps...>(),
120 [&result](auto&& pair)
121 {
122 const auto& second = boost::hana::second(pair);
123 boost::hana::for_each(second,
124 [&result](auto&& pair2)
125 {
126 const auto second2 = boost::hana::second(pair2);
127 result += second2->size();
128 });
129 });
130 return result;
131}
