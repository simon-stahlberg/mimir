343template<IsVertex V, IsEdge E>
344template<typename... VertexProperties>
345 requires HasVertexProperties<V, VertexProperties...>
346VertexIndex StaticGraph<V, E>::add_vertex(VertexProperties&&... properties)
347{
348 const auto index = m_vertices.size();
349 m_vertices.emplace_back(index, std::forward<VertexProperties>(properties)...);
350
351 boost::hana::at_key(m_degrees, boost::hana::type<Forward> {}).resize(index + 1, 0);
352 boost::hana::at_key(m_degrees, boost::hana::type<Backward> {}).resize(index + 1, 0);
353
354 return index;
355}

357template<IsVertex V, IsEdge E>
358template<typename... VertexProperties>
359 requires HasVertexProperties<V, VertexProperties...>
360VertexIndex StaticGraph<V, E>::add_vertex(const Vertex<VertexProperties...>& vertex)
361{
362 return std::apply([this](auto&&... properties) { return this->add_vertex(std::forward<decltype(properties)>(properties)...); }, vertex.get_properties());
363}

365template<IsVertex V, IsEdge E>
366template<typename... EdgeProperties>
367 requires HasEdgeProperties<E, EdgeProperties...>
368EdgeIndex StaticGraph<V, E>::add_directed_edge(VertexIndex source, VertexIndex target, EdgeProperties&&... properties)
369{
370 vertex_index_check(source, "StaticGraph<V, E>::add_directed_edge(...): Source vertex out of range");
371 vertex_index_check(target, "StaticGraph<V, E>::add_directed_edge(...): Source vertex out of range");
372
373 const auto index = m_edges.size();
374 m_edges.emplace_back(index, source, target, std::forward<EdgeProperties>(properties)...);
375 ++boost::hana::at_key(m_degrees, boost::hana::type<Forward> {}).at(source);
376 ++boost::hana::at_key(m_degrees, boost::hana::type<Backward> {}).at(target);
377 m_slice.push_back(index);
378
379 return index;
380}

382template<IsVertex V, IsEdge E>
383template<typename... EdgeProperties>
384 requires HasEdgeProperties<E, EdgeProperties...>
385EdgeIndex StaticGraph<V, E>::add_directed_edge(VertexIndex source, VertexIndex target, const Edge<EdgeProperties...>& edge)
386{
387 return std::apply([this, source, target](auto&&... properties)
388 { return this->add_directed_edge(source, target, std::forward<decltype(properties)>(properties)...); },
389 edge.get_properties());
390}

392template<IsVertex V, IsEdge E>
393template<typename... EdgeProperties>
394 requires HasEdgeProperties<E, EdgeProperties...>
395std::pair<EdgeIndex, EdgeIndex> StaticGraph<V, E>::add_undirected_edge(VertexIndex source, VertexIndex target, EdgeProperties&&... properties)
396{
397 auto properties_tuple = std::make_tuple(std::forward<EdgeProperties>(properties)...);
398 auto properties_tuple_copy = properties_tuple;
399
400 const auto forward_edge_index =
401 std::apply([this, source, target](auto&&... args) { return this->add_directed_edge(source, target, std::forward<decltype(args)>(args)...); },
402 std::move(properties_tuple_copy));
403 const auto backward_edge_index =
404 std::apply([this, source, target](auto&&... args) { return this->add_directed_edge(target, source, std::forward<decltype(args)>(args)...); },
405 std::move(properties_tuple));
406
407 return std::make_pair(forward_edge_index, backward_edge_index);
408}

602template<IsVertex V, IsEdge E>
603static IndexGroupedVector<const EdgeIndex> compute_index_grouped_edge_indices(const StaticGraph<V, E>& graph, bool forward)
604{
605 auto grouped_by_source_data = std::vector<std::pair<VertexIndex, EdgeIndex>> {};
606 for (const auto& edge : graph.get_edges())
607 {
608 grouped_by_source_data.emplace_back(((forward) ? edge.get_source() : edge.get_target()), edge.get_index());
609 }
610
611 std::sort(grouped_by_source_data.begin(), grouped_by_source_data.end(), [](const auto& l, const auto& r) { return l.first < r.first; });
612
613 auto index_grouped_by_source = IndexGroupedVector<std::pair<VertexIndex, EdgeIndex>>::create(
614 std::move(grouped_by_source_data),
615 [](const auto& l, const auto& r) { return l.first != r.first; },
616 [](const auto& e) { return e.first; },
617 graph.get_num_vertices());
618
619 auto edge_index_grouped_by_source_builder = IndexGroupedVectorBuilder<const EdgeIndex>();
620 for (const auto& group : index_grouped_by_source)
621 {
622 edge_index_grouped_by_source_builder.start_group();
623
624 for (const auto& element : group)
625 {
626 edge_index_grouped_by_source_builder.add_group_element(element.second);
627 }
628 }
629 return edge_index_grouped_by_source_builder.get_result();
630}
