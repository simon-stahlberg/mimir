358template<IsVertex V, IsEdge E>
359template<typename... VertexProperties>
360 requires HasVertexProperties<V, VertexProperties...>
361VertexIndex StaticGraph<V, E>::add_vertex(VertexProperties&&... properties)
362{
363 const auto index = m_vertices.size();
364 m_vertices.emplace_back(index, std::forward<VertexProperties>(properties)...);
365
366 boost::hana::at_key(m_degrees, boost::hana::type<ForwardTraversal> {}).resize(index + 1, 0);
367 boost::hana::at_key(m_degrees, boost::hana::type<BackwardTraversal> {}).resize(index + 1, 0);
368
369 return index;
370}

372template<IsVertex V, IsEdge E>
373template<typename... VertexProperties>
374 requires HasVertexProperties<V, VertexProperties...>
375VertexIndex StaticGraph<V, E>::add_vertex(const Vertex<VertexProperties...>& vertex)
376{
377 return std::apply([this](auto&&... properties) { return this->add_vertex(std::forward<decltype(properties)>(properties)...); }, vertex.get_properties());
378}

380template<IsVertex V, IsEdge E>
381template<typename... EdgeProperties>
382 requires HasEdgeProperties<E, EdgeProperties...>
383EdgeIndex StaticGraph<V, E>::add_directed_edge(VertexIndex source, VertexIndex target, EdgeProperties&&... properties)
384{
385 vertex_index_check(source, "StaticGraph<V, E>::add_directed_edge(...): Source vertex out of range");
386 vertex_index_check(target, "StaticGraph<V, E>::add_directed_edge(...): Source vertex out of range");
387
388 const auto index = m_edges.size();
389 m_edges.emplace_back(index, source, target, std::forward<EdgeProperties>(properties)...);
390 ++boost::hana::at_key(m_degrees, boost::hana::type<ForwardTraversal> {}).at(source);
391 ++boost::hana::at_key(m_degrees, boost::hana::type<BackwardTraversal> {}).at(target);
392 m_slice.push_back(index);
393
394 return index;
395}

397template<IsVertex V, IsEdge E>
398template<typename... EdgeProperties>
399 requires HasEdgeProperties<E, EdgeProperties...>
400EdgeIndex StaticGraph<V, E>::add_directed_edge(VertexIndex source, VertexIndex target, const Edge<EdgeProperties...>& edge)
401{
402 return std::apply([this, source, target](auto&&... properties)
403 { return this->add_directed_edge(source, target, std::forward<decltype(properties)>(properties)...); },
404 edge.get_properties());
405}

407template<IsVertex V, IsEdge E>
408template<typename... EdgeProperties>
409 requires HasEdgeProperties<E, EdgeProperties...>
410std::pair<EdgeIndex, EdgeIndex> StaticGraph<V, E>::add_undirected_edge(VertexIndex source, VertexIndex target, EdgeProperties&&... properties)
411{
412 auto properties_tuple = std::make_tuple(std::forward<EdgeProperties>(properties)...);
413 auto properties_tuple_copy = properties_tuple;
414
415 const auto forward_edge_index =
416 std::apply([this, source, target](auto&&... args) { return this->add_directed_edge(source, target, std::forward<decltype(args)>(args)...); },
417 std::move(properties_tuple_copy));
418 const auto backward_edge_index =
419 std::apply([this, source, target](auto&&... args) { return this->add_directed_edge(target, source, std::forward<decltype(args)>(args)...); },
420 std::move(properties_tuple));
421
422 return std::make_pair(forward_edge_index, backward_edge_index);
423}

617template<IsVertex V, IsEdge E>
618static IndexGroupedVector<const EdgeIndex> compute_index_grouped_edge_indices(const StaticGraph<V, E>& graph, bool forward)
619{
620 auto grouped_by_source_data = std::vector<std::pair<VertexIndex, EdgeIndex>> {};
621 for (const auto& edge : graph.get_edges())
622 {
623 grouped_by_source_data.emplace_back(((forward) ? edge.get_source() : edge.get_target()), edge.get_index());
624 }
625
626 std::sort(grouped_by_source_data.begin(), grouped_by_source_data.end(), [](const auto& l, const auto& r) { return l.first < r.first; });
627
628 auto index_grouped_by_source = IndexGroupedVector<std::pair<VertexIndex, EdgeIndex>>::create(
629 std::move(grouped_by_source_data),
630 [](const auto& l, const auto& r) { return l.first != r.first; },
631 [](const auto& e) { return e.first; },
632 graph.get_num_vertices());
633
634 auto edge_index_grouped_by_source_builder = IndexGroupedVectorBuilder<const EdgeIndex>();
635 for (const auto& group : index_grouped_by_source)
636 {
637 edge_index_grouped_by_source_builder.start_group();
638
639 for (const auto& element : group)
640 {
641 edge_index_grouped_by_source_builder.add_group_element(element.second);
642 }
643 }
644 return edge_index_grouped_by_source_builder.get_result();
645}
