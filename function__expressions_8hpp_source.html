250inline ContinuousCost evaluate_binary(loki::BinaryOperatorEnum op, ContinuousCost val_left, ContinuousCost val_right)
251{
252 if (val_left == UNDEFINED_CONTINUOUS_COST || val_right == UNDEFINED_CONTINUOUS_COST)
253 {
254 return UNDEFINED_CONTINUOUS_COST;
255 }
256
257 switch (op)
258 {
259 case loki::BinaryOperatorEnum::DIV:
260 {
261 if (val_right == 0.)
262 return UNDEFINED_CONTINUOUS_COST;
263
264 return val_left / val_right;
265 }
266 case loki::BinaryOperatorEnum::MINUS:
267 {
268 return val_left - val_right;
269 }
270 case loki::BinaryOperatorEnum::MUL:
271 {
272 return val_left * val_right;
273 }
274 case loki::BinaryOperatorEnum::PLUS:
275 {
276 return val_left + val_right;
277 }
278 default:
279 {
280 throw std::logic_error("Evaluation of binary operator is undefined.");
281 }
282 }
283}

285inline ContinuousCost evaluate_multi(loki::MultiOperatorEnum op, ContinuousCost val_left, ContinuousCost val_right)
286{
287 if (val_left == UNDEFINED_CONTINUOUS_COST || val_right == UNDEFINED_CONTINUOUS_COST)
288 {

309template<IsArithmetic A>
310inline Bounds<A> evaluate_binary_bounds(loki::BinaryOperatorEnum op, const Bounds<A>& lhs, const Bounds<A>& rhs)
311{
312 const auto alternative1 = evaluate_binary(op, lhs.get_lower(), rhs.get_lower());
313 const auto alternative2 = evaluate_binary(op, lhs.get_upper(), rhs.get_lower());
314 const auto alternative3 = evaluate_binary(op, lhs.get_lower(), rhs.get_upper());
315 const auto alternative4 = evaluate_binary(op, lhs.get_upper(), rhs.get_upper());
316 auto result = std::minmax({ alternative1, alternative2, alternative3, alternative4 });
317 return Bounds<A>(result.first, result.second);
318}

320template<IsArithmetic A>
321inline Bounds<A> evaluate_multi_bounds(loki::MultiOperatorEnum op, const Bounds<A>& lhs, const Bounds<A>& rhs)
322{
323 const auto alternative1 = evaluate_multi(op, lhs.get_lower(), rhs.get_lower());
324 const auto alternative2 = evaluate_multi(op, lhs.get_upper(), rhs.get_lower());
325 const auto alternative3 = evaluate_multi(op, lhs.get_lower(), rhs.get_upper());
326 const auto alternative4 = evaluate_multi(op, lhs.get_upper(), rhs.get_upper());
327 auto result = std::minmax({ alternative1, alternative2, alternative3, alternative4 });
328 return Bounds<A>(result.first, result.second);
329}
