49template<IsStaticGraph Graph, IsDirection Direction>
50 requires IsVertexListGraph<Graph> && IsIncidenceGraph<Graph>
51std::pair<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type,
52 std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type>>
53strong_components(const DirectionTaggedType<Graph, Direction>& g)
54{
55 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
56 using vertices_size_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type;
57
58 auto vertex_index_map = TrivialReadPropertyMap<vertex_descriptor_type, vertex_descriptor_type>();
59 auto c = std::vector<vertices_size_type>(g.get().get_num_vertices());
60 auto component_map = VectorReadWritePropertyMap<vertex_descriptor_type, vertices_size_type>(c);
61
62 const auto num_components = boost::strong_components(g, component_map, boost::vertex_index_map(vertex_index_map));
63 return std::make_pair(num_components, c);
64}

92template<IsStaticGraph Graph, IsDirection Direction, class SourceInputIter>
93 requires IsVertexListGraph<Graph> && IsIncidenceGraph<Graph>
94std::tuple<std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor>, ContinuousCostList>
95dijkstra_shortest_paths(const DirectionTaggedType<Graph, Direction>& g, const ContinuousCostList& w, SourceInputIter s_begin, SourceInputIter s_end)
96{
97 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
98 using edge_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::edge_descriptor;
99
100 auto p = std::vector<vertex_descriptor_type>(g.get().get_num_vertices());
101 auto predecessor_map = VectorReadWritePropertyMap<vertex_descriptor_type, vertex_descriptor_type>(p);
102 auto d = ContinuousCostList(g.get().get_num_vertices());
103 auto distance_map = VectorReadWritePropertyMap<vertex_descriptor_type, ContinuousCost>(d);
104 auto weight_map = VectorReadPropertyMap<edge_descriptor_type, ContinuousCost>(w);
105 auto vertex_index_map = TrivialReadPropertyMap<vertex_descriptor_type, vertex_descriptor_type>();
106 auto compare = std::less<ContinuousCost>();
107 auto combine = std::plus<ContinuousCost>();
108 auto inf = std::numeric_limits<ContinuousCost>::infinity();
109 auto zero = ContinuousCost();
110
111 // multiple source shortest path version.
112 boost::dijkstra_shortest_paths(g, //
113 s_begin,
114 s_end,
115 predecessor_map,
116 distance_map,
117 weight_map,
118 vertex_index_map,
119 compare,
120 combine,
121 inf,
122 zero,
123 boost::default_dijkstra_visitor());
124
125 return std::make_tuple(p, d);
126};

132template<typename VertexDescriptor>
133struct BFSBoostVisitor : public boost::bfs_visitor<>
134{
135 std::reference_wrapper<std::vector<VertexDescriptor>> predecessors;
136 std::reference_wrapper<DiscreteCostList> distances;
137
138 BFSBoostVisitor(std::vector<VertexDescriptor>& p, DiscreteCostList& d) : predecessors(p), distances(d) {}

140 template<typename Edge, typename Graph>
141 void tree_edge(Edge e, const Graph& g) const
142 {
143 auto u = source(e, g);
144 auto v = target(e, g);
145 predecessors.get().at(v) = u;
146 distances.get().at(v) = distances.get().at(u) + 1;
147 }
148};

150template<IsStaticGraph Graph, IsDirection Direction, class SourceInputIter>
151std::tuple<std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor>, DiscreteCostList>
152breadth_first_search(const DirectionTaggedType<Graph, Direction>& g, SourceInputIter s_begin, SourceInputIter s_end)
153{
154 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
155 using ColorMap = boost::iterator_property_map<std::vector<boost::default_color_type>::iterator, boost::identity_property_map>;
156
157 auto buffer = boost::queue<vertex_descriptor_type>();
158 auto p = std::vector<vertex_descriptor_type>(g.get().get_num_vertices());
159 for (vertex_descriptor_type v = 0; v < g.get().get_num_vertices(); ++v)
160 {
161 p.at(v) = v;
162 }
163 auto inf = std::numeric_limits<DiscreteCost>::max();
164 auto d = DiscreteCostList(g.get().get_num_vertices(), inf);
165 for (auto it = s_begin; it != s_end; ++it)
166 {
167 d.at(*it) = DiscreteCost(0);
168 }
169 auto visitor = BFSBoostVisitor(p, d);
170 auto color_vector = std::vector<boost::default_color_type>(g.get().get_num_vertices(), boost::white_color);
171 auto color_map = ColorMap(color_vector.begin(), boost::identity_property_map());
172 boost::breadth_first_search(g, s_begin, s_end, buffer, visitor, color_map);
173
174 return std::make_tuple(p, d);
175}
