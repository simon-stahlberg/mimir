43template<IsStaticGraph Graph, IsDirection Direction, class SourceInputIter>
44 requires IsVertexListGraph<Graph> && IsIncidenceGraph<Graph>
45auto dijkstra_shortest_paths(const DirectionTaggedType<Graph, Direction>& g, const ContinuousCostList& w, SourceInputIter s_begin, SourceInputIter s_end)
46{
47 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
48 using edge_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::edge_descriptor;
49
50 auto p = std::vector<vertex_descriptor_type>(g.get().get_num_vertices());
51 auto predecessor_map = VectorReadWritePropertyMap<vertex_descriptor_type, vertex_descriptor_type>(p);
52 auto d = ContinuousCostList(g.get().get_num_vertices());
53 auto distance_map = VectorReadWritePropertyMap<vertex_descriptor_type, ContinuousCost>(d);
54 auto weight_map = VectorReadPropertyMap<edge_descriptor_type, ContinuousCost>(w);
55 auto vertex_index_map = TrivialReadPropertyMap<vertex_descriptor_type, vertex_descriptor_type>();
56 auto compare = std::less<ContinuousCost>();
57 auto combine = std::plus<ContinuousCost>();
58 auto inf = std::numeric_limits<ContinuousCost>::infinity();
59 auto zero = ContinuousCost();
60
61 // multiple source shortest path version.
62 boost::dijkstra_shortest_paths(g, //
63 s_begin,
64 s_end,
65 predecessor_map,
66 distance_map,
67 weight_map,
68 vertex_index_map,
69 compare,
70 combine,
71 inf,
72 zero,
73 boost::default_dijkstra_visitor());
74
75 return std::make_tuple(p, d);
76};

82template<IsStaticGraph Graph, IsDirection Direction>
83 requires IsVertexListGraph<Graph> && IsEdgeListGraph<Graph>
84auto floyd_warshall_all_pairs_shortest_paths(const DirectionTaggedType<Graph, Direction>& g, const ContinuousCostList& w)
85{
86 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
87 using edge_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::edge_descriptor;
88
89 using BasicMatrix = typename PropertyMapTraits<std::decay_t<Graph>>::template BasicMatrix<vertex_descriptor_type, ContinuousCost>;
90 using WeightPropertyMap = typename PropertyMapTraits<std::decay_t<Graph>>::template ReadPropertyMap<edge_descriptor_type, ContinuousCost>;
91
92 auto matrix = BasicMatrix::initialize_data(g.get().get_num_vertices());
93
94 auto d = BasicMatrix(matrix);
95 auto weight_map = WeightPropertyMap(w);
96
97 boost::floyd_warshall_all_pairs_shortest_paths(g, d, boost::weight_map(weight_map));
98
99 return matrix;
100}
