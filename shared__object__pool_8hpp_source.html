34template<typename T>
35class SharedObjectPoolPtr
36{
37private:
38 SharedObjectPool<T>* m_pool;
39 std::pair<size_t, T>* m_object;
40
41private:
42 void deallocate()
43 {
44 assert(m_pool && m_object);
45
46 m_pool->free(m_object);
47 m_pool = nullptr;
48 m_object = nullptr;
49 }
50
51 std::pair<size_t, T>* release() noexcept
52 {
53 std::pair<size_t, T>* temp = m_object;
54 m_object = nullptr;
55 m_pool = nullptr;
56 return temp;
57 }
58
59 void inc_ref_count()
60 {
61 assert(m_object);
62
63 ++m_object->first;
64 }
65
66 void dec_ref_count()
67 {
68 assert(m_object);
69 assert(m_object->first > 0);
70
71 --m_object->first;
72
73 if (m_object->first == 0)
74 {
75 deallocate();
76 }
77 }
78
79public:
80 SharedObjectPoolPtr() : SharedObjectPoolPtr(nullptr, nullptr) {}

184template<typename T>
185class SharedObjectPool
186{
187private:
188 using Entry = std::pair<size_t, T>;
189
190 std::vector<std::unique_ptr<Entry>> m_storage;
191 std::stack<Entry*> m_stack;
192
193 template<typename... Args>
194 void allocate(Args&&... args)
195 {
196 m_storage.push_back(std::make_unique<Entry>(size_t(0), T(std::forward<Args>(args)...)));
197 m_stack.push(m_storage.back().get());
198 }
199
200 void free(Entry* element) { m_stack.push(element); }
201
202 friend class SharedObjectPoolPtr<T>;
203
204public:
205 // Non-copyable to prevent dangling memory pool pointers.
206 SharedObjectPool() = default;
207 SharedObjectPool(const SharedObjectPool& other) = delete;
208 SharedObjectPool& operator=(const SharedObjectPool& other) = delete;
209 SharedObjectPool(SharedObjectPool&& other) = delete;
210 SharedObjectPool& operator=(SharedObjectPool&& other) = delete;
211
212 [[nodiscard]] SharedObjectPoolPtr<T> get_or_allocate() { return get_or_allocate<>(); }
213
214 template<typename... Args>
215 [[nodiscard]] SharedObjectPoolPtr<T> get_or_allocate(Args&&... args)
216 {
217 if (m_stack.empty())
218 {
219 allocate(std::forward<Args>(args)...);
220 }
221 Entry* element = m_stack.top();
222 m_stack.pop();
223 return SharedObjectPoolPtr<T>(this, element);
224 }
225
226 [[nodiscard]] size_t get_size() const { return m_storage.size(); }
227
228 [[nodiscard]] size_t get_num_free() const { return m_stack.size(); }
229};
