43template<std::unsigned_integral I, typename V>
44class VectorBasicMatrix
45{
46public:
47 VectorBasicMatrix(std::vector<std::vector<V>>& matrix) : m_matrix(matrix) {}
48
49 static std::vector<std::vector<V>> initialize_data(std::size_t num_vertices)
50 {
51 return std::vector<std::vector<V>>(num_vertices, std::vector<V>(num_vertices));
52 }
53
54 std::vector<V>& operator[](I i) { return m_matrix.get()[i]; }
55
56 const std::vector<V>& operator[](I i) const { return m_matrix.get()[i]; }
57
58 const std::vector<std::vector<V>>& get_matrix() const { return m_matrix.get(); }
59
60private:
61 std::reference_wrapper<std::vector<std::vector<V>>> m_matrix;
62};
