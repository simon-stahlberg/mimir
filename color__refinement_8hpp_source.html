94template<typename ColorType>
95void replace_tuples(const std::vector<std::pair<Index, ColorType>>& M,
96 const ColorIndexList& hash_to_color,
97 std::vector<std::tuple<ColorIndex, std::vector<ColorType>, Index>>& out_M_replaced)
98{
99 // Subroutine to construct signatures.
100 auto it = M.begin();
101 while (it != M.end())
102 {
103 auto signature = std::vector<ColorType>();
104 const auto hash = it->first;
105
106 auto it2 = it;
107 while (it2 != M.end() && it2->first == hash)
108 {
109 signature.push_back(it2->second);
110 ++it2;
111 }
112 it = it2;
113
114 // Ensure canonical signature.
115 assert(std::is_sorted(signature.begin(), signature.end()));
116 out_M_replaced.emplace_back(hash_to_color.at(hash), std::move(signature), hash);
117 }
118}

128template<typename ColorType>
129void split_color_classes(const std::vector<std::tuple<ColorIndex, std::vector<ColorType>, Index>>& M_replaced,
130 UnorderedMap<std::pair<ColorIndex, std::vector<ColorType>>, ColorIndex>& ref_f,
131 ColorIndex& ref_max_color,
132 ColorIndexList& ref_hash_to_color,
133 ColorIndexMap<IndexList>& out_color_to_hashes,
134 ColorIndexSet& out_L)
135{
136 out_color_to_hashes.clear();
137 out_L.clear();
138
139 // Subroutine to split color classes.
140 auto it = M_replaced.begin();
141 while (it != M_replaced.end())
142 {
143 const auto old_color = std::get<0>(*it);
144
145 /* Check whether old color has split. */
146 auto has_split = false;
147 auto it2 = it;
148 {
149 // Subroutine to detect split
150 const auto& signature = std::get<1>(*it);
151 while (it2 != M_replaced.end() && old_color == std::get<0>(*it2))
152 {
153 if (signature != std::get<1>(*it2))
154 {
155 has_split = true;
156 break;
157 }
158 ++it2;
159 }
160 }
161
162 if (!has_split)
163 {
164 /* Move to next old color class or the end. */
165 it = it2;
166 }
167 else
168 {
169 /* Split old_color class. */
170 {
171 while (it != M_replaced.end() && old_color == std::get<0>(*it))
172 {
173 // Determine new color for (old_color, signature)
174 const auto& signature = std::get<1>(*it);
175 const auto new_color = ++ref_max_color;
176
177 // Add new color to work list.
178 out_L.insert(new_color);
179
180 // Add mapping to decoding table
181 [[maybe_unused]] const auto result = ref_f.emplace(std::make_pair(old_color, signature), new_color);
182 // Ensure that we are not overwritting table entries.
183 assert(result.second);
184
185 {
186 // Subroutine to assign new color to vertices with same signature.
187 while (it != M_replaced.end() && old_color == std::get<0>(*it) && signature == std::get<1>(*it))
188 {
189 auto hash = std::get<2>(*it);
190 ref_hash_to_color[hash] = new_color;
191 out_color_to_hashes[new_color].push_back(hash);
192 ++it;
193 }
194 }
195 }
196 }
197 }
198 }

201template<typename G>
202 requires IsVertexListGraph<G> && IsIncidenceGraph<G> && IsVertexColoredGraph<G> //
203std::shared_ptr<CertificateImpl> compute_certificate(const G& graph)
204{
205 if (!is_undirected_graph(graph))
206 {
207 throw std::runtime_error("Color-refinement does not support directed graphs because they can be translated into undirected graphs by introducing two "
208 "vertices along the edge with different colors to encode the direction.");
209 }
210
211 // Toggle verbosity
212 const bool debug = false;
213
214 /* Fetch some data. */
215 const auto num_vertices = graph.get_num_vertices();
216
217 /* Canonize and compress the IColors to a list of integers. */
218 auto c = CertificateImpl::ColorCompressionFunction {};
219 auto canonical_coloring = ColorList {};
220 for (const auto& vertex : graph.get_vertices())
221 {
222 canonical_coloring.push_back(get_color(vertex));
223 }
224 std::sort(canonical_coloring.begin(), canonical_coloring.end());
225 for (const auto& color : canonical_coloring)
226 {
227 c.emplace(color, c.size());
228 }
229
230 // (line 1-2): Initialize vertex colors and perfect hashes for vertex indices.
231 auto vertex_to_hash = IndexMap<Index>();
232 auto hash_to_vertex = IndexMap<Index>();
233 auto max_color = ColorIndex();
234 auto hash_to_color = ColorIndexList(num_vertices);
235 auto color_to_hashes = ColorIndexMap<IndexList>();
236 // (line 1-2): Initialize work list.
237 auto L = ColorIndexSet();
238 for (const auto& vertex : graph.get_vertices())
239 {
240 const auto hash = vertex_to_hash.size();
241 vertex_to_hash.emplace(vertex.get_index(), hash);
242 hash_to_vertex.emplace(hash, vertex.get_index());
243
244 const auto color = c.at(get_color(vertex));
245 max_color = std::max(max_color, color);
246 hash_to_color[hash] = color;
247 color_to_hashes[color].push_back(hash);
248 L.insert(color);
249 }
250
251 // (line 1-2): Initialize multi set.
252 auto f = CertificateImpl::CompressionFunction();
253 auto M = std::vector<std::pair<Index, ColorIndex>>();
254 auto M_replaced = std::vector<std::tuple<ColorIndex, ColorIndexList, Index>>();
255 // (line 3): Process work list until all vertex colors have stabilized.
256 while (!L.empty())
257 {
258 if (debug)
259 {
260 std::cout << "L: ";
261 mimir::operator<<(std::cout, L);
262 std::cout << std::endl;
263 }
264
265 // Clear data structures that are reused.
266 M.clear();
267 M_replaced.clear();
268
269 {
270 // Subroutine to compute multiset M.
271 // Note: this computes the stable coloring, not the coarsest stable coloring.
272 for (size_t h = 0; h < num_vertices; ++h)
273 {
274 for (const auto& outgoing_vertex : graph.template get_adjacent_vertices<ForwardTag>(hash_to_vertex.at(h)))
275 {
276 const auto hash = vertex_to_hash.at(outgoing_vertex.get_index());
277
278 M.emplace_back(h, hash_to_color.at(hash));
279 }
280 }
281 }
282
283 // (line 12): Perform radix sort of M by vertex and colors (TODO radix sort)
284 std::sort(M.begin(), M.end());
285
286 if (debug)
287 {
288 std::cout << "M: ";
289 mimir::operator<<(std::cout, M);
290 std::cout << std::endl;
291 }
292
293 // (line 13): Scan M and replace tuples (v,c1),...,(v,cr) with single tuple (C(v),c1,...,cr,v) to construct signatures.
294 replace_tuples(M, hash_to_color, M_replaced);
295
296 // (line 14): Perform radix sort of M by old color and neighborhood colors (TODO radix sort)
297 std::sort(M_replaced.begin(), M_replaced.end());
298
299 if (debug)
300 {
301 std::cout << "M_replaced: ";
302 mimir::operator<<(std::cout, M_replaced);
303 std::cout << std::endl;
304 }
305
306 /* (line 15): Add new colors to work list */
307 split_color_classes(M_replaced, f, max_color, hash_to_color, color_to_hashes, L);
308 }
309
310 /* Report final neighborhood structures in the decoding table. */
311 for (const auto& [old_color, signature, hash] : M_replaced)
312 {
313 f.emplace(std::make_pair(old_color, signature), old_color);
314 }
315
316 /* Return the certificate */
317 return std::make_shared<CertificateImpl>(std::move(c), std::move(f));
318}
