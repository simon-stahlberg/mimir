233template<IsStaticGraph Graph, IsDirection Direction>
234std::pair<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type,
235 std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type>>
236strong_components(const DirectionTaggedType<Graph, Direction>& g)
237{
238 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
239 using vertices_size_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type;
240
241 auto vertex_index_map = TrivialReadPropertyMap<vertex_descriptor_type, vertex_descriptor_type>();
242 auto c = std::vector<vertices_size_type>(g.get().get_num_vertices());
243 auto component_map = VectorReadWritePropertyMap<vertex_descriptor_type, vertices_size_type>(c);
244
245 const auto num_components = boost::strong_components(g, component_map, boost::vertex_index_map(vertex_index_map));
246 return std::make_pair(num_components, c);
247}

249template<IsStaticGraph Graph, IsDirection Direction>
250IndexGroupedVector<std::pair<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type,
251 typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor>>
252get_partitioning(typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type num_components,
253 std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type> component_map)
254{
255 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
256 using vertices_size_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type;
257 using state_component_pair_t = std::pair<vertices_size_type, vertex_descriptor_type>;
258
259 auto partitioning = std::vector<state_component_pair_t>();
260 for (vertex_descriptor_type v = 0; v < component_map.size(); ++v)
261 {
262 partitioning.push_back({ component_map.at(v), v });
263 }
264 std::sort(std::begin(partitioning), std::end(partitioning));
265 return IndexGroupedVector<state_component_pair_t>::create(std::move(partitioning),
266 [](const auto& prev, const auto& cur) { return prev.first != cur.first; });
267}

278template<IsStaticGraph Graph, IsDirection Direction, class SourceInputIter>
279std::tuple<std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor>, ContinuousCostList>
280dijkstra_shortest_paths(const DirectionTaggedType<Graph, Direction>& g, const ContinuousCostList& w, SourceInputIter s_begin, SourceInputIter s_end)
281{
282 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
283 using edge_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::edge_descriptor;
284
285 auto p = std::vector<vertex_descriptor_type>(g.get().get_num_vertices());
286 auto predecessor_map = VectorReadWritePropertyMap<vertex_descriptor_type, vertex_descriptor_type>(p);
287 auto d = ContinuousCostList(g.get().get_num_vertices());
288 auto distance_map = VectorReadWritePropertyMap<vertex_descriptor_type, ContinuousCost>(d);
289 auto weight_map = VectorReadPropertyMap<edge_descriptor_type, ContinuousCost>(w);
290 auto vertex_index_map = TrivialReadPropertyMap<vertex_descriptor_type, vertex_descriptor_type>();
291 auto compare = std::less<ContinuousCost>();
292 auto combine = std::plus<ContinuousCost>();
293 auto inf = std::numeric_limits<ContinuousCost>::infinity();
294 auto zero = ContinuousCost();
295
296 // multiple source shortest path version.
297 boost::dijkstra_shortest_paths(g, //
298 s_begin,
299 s_end,
300 predecessor_map,
301 distance_map,
302 weight_map,
303 vertex_index_map,
304 compare,
305 combine,
306 inf,
307 zero,
308 boost::default_dijkstra_visitor());
309
310 return std::make_tuple(p, d);
311};

317template<typename VertexDescriptor>
318struct BFSBoostVisitor : public boost::bfs_visitor<>
319{
320 std::reference_wrapper<std::vector<VertexDescriptor>> predecessors;
321 std::reference_wrapper<DiscreteCostList> distances;
322
323 BFSBoostVisitor(std::vector<VertexDescriptor>& p, DiscreteCostList& d) : predecessors(p), distances(d) {}
324
325 template<typename Edge, typename Graph>
326 void tree_edge(Edge e, const Graph& g) const
327 {
328 auto u = source(e, g);
329 auto v = target(e, g);
330 predecessors.get().at(v) = u;
331 distances.get().at(v) = distances.get().at(u) + 1;
332 }
333};

335template<IsStaticGraph Graph, IsDirection Direction, class SourceInputIter>
336std::tuple<std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor>, DiscreteCostList>
337breadth_first_search(const DirectionTaggedType<Graph, Direction>& g, SourceInputIter s_begin, SourceInputIter s_end)
338{
339 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
340 using ColorMap = boost::iterator_property_map<std::vector<boost::default_color_type>::iterator, boost::identity_property_map>;
341
342 auto buffer = boost::queue<vertex_descriptor_type>();
343 auto p = std::vector<vertex_descriptor_type>(g.get().get_num_vertices());
344 for (vertex_descriptor_type v = 0; v < g.get().get_num_vertices(); ++v)
345 {
346 p.at(v) = v;
347 }
348 auto inf = std::numeric_limits<DiscreteCost>::max();
349 auto d = DiscreteCostList(g.get().get_num_vertices(), inf);
350 for (auto it = s_begin; it != s_end; ++it)
351 {
352 d.at(*it) = DiscreteCost(0);
353 }
354 auto visitor = BFSBoostVisitor(p, d);
355 auto color_vector = std::vector<boost::default_color_type>(g.get().get_num_vertices(), boost::white_color);
356 auto color_map = ColorMap(color_vector.begin(), boost::identity_property_map());
357 boost::breadth_first_search(g, s_begin, s_end, buffer, visitor, color_map);
358
359 return std::make_tuple(p, d);
360}
