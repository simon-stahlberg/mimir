157 self().on_generate_state_relaxed_impl(state, action, action_cost, successor_state);
158 }

161 void on_close_state(const State& state) override
162 {
163 if (!m_quiet)
164 {
165 self().on_close_state_impl(state);
166 }
167 }
168
169 void on_finish_f_layer(ContinuousCost f_value) override
170 {
171 m_statistics.on_finish_f_layer(f_value);
172
173 if (!m_quiet)
174 {
175 assert(!m_statistics.get_num_expanded_until_f_value().empty());
176 self().on_finish_f_layer_impl(f_value,
177 m_statistics.get_num_expanded_until_f_value().rbegin()->second,
178 m_statistics.get_num_generated_until_f_value().rbegin()->second);
179 }
180 }
181
182 void on_prune_state(const State& state) override
183 {
184 m_statistics.increment_num_pruned();
185
186 if (!m_quiet)
187 {
188 self().on_prune_state_impl(state);
189 }
190 }
191
192 void on_start_search(const State& start_state, ContinuousCost g_value, ContinuousCost h_value) override
193 {
194 m_statistics = Statistics();
195
196 m_statistics.set_search_start_time_point(std::chrono::high_resolution_clock::now());
197
198 if (!m_quiet)
199 {
200 self().on_start_search_impl(start_state, g_value, h_value);
201 }
202 }
203
204 void on_end_search(uint64_t num_reached_fluent_atoms,
205 uint64_t num_reached_derived_atoms,
206 uint64_t num_states,
207 uint64_t num_nodes,
208 uint64_t num_actions,
209 uint64_t num_axioms) override
210
211 {
212 m_statistics.set_search_end_time_point(std::chrono::high_resolution_clock::now());
213 m_statistics.set_num_reached_fluent_atoms(num_reached_fluent_atoms);
214 m_statistics.set_num_reached_derived_atoms(num_reached_derived_atoms);
215 m_statistics.set_num_states(num_states);
216 m_statistics.set_num_nodes(num_nodes);
217 m_statistics.set_num_actions(num_actions);
218 m_statistics.set_num_axioms(num_axioms);
219
220 if (!m_quiet)
221 {
222 self().on_end_search_impl(num_reached_fluent_atoms, num_reached_derived_atoms, num_states, num_nodes, num_actions, num_axioms);
223 }
224 }
225
226 void on_solved(const Plan& plan) override
227 {
228 if (!m_quiet)
229 {
230 self().on_solved_impl(plan);
231 }
232 }
233
234 void on_unsolvable() override
235 {
236 if (!m_quiet)
237 {
238 self().on_unsolvable_impl();
239 }
240 }
241
242 void on_exhausted() override
243 {
244 if (!m_quiet)
245 {
246 self().on_exhausted_impl();
247 }
248 }
249
254 const Statistics& get_statistics() const override { return m_statistics; }
255 bool is_quiet() const { return m_quiet; }
256};
