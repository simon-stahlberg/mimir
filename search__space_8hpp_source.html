35template<IsSearchNode SearchNode>
36IndexList extract_state_trajectory(const SegmentedVector<SearchNode>& search_nodes, const SearchNode& final_search_node, Index final_state_index)
37{
38 auto trajectory = IndexList {};
39 trajectory.push_back(final_state_index);
40
41 auto cur_search_node = &final_search_node;
42
43 while (cur_search_node->parent_state != std::numeric_limits<Index>::max())
44 {
45 trajectory.push_back(cur_search_node->parent_state);
46
47 cur_search_node = &search_nodes.at(cur_search_node->parent_state);
48 }
49
50 std::reverse(trajectory.begin(), trajectory.end());
51
52 return trajectory;
53}

63template<IsSearchNode SearchNode>
64inline Plan extract_total_ordered_plan(State start_state,
65 ContinuousCost start_state_metric_value,
66 const SearchNode& final_search_node,
67 Index final_state_index,
68 const SegmentedVector<SearchNode>& search_nodes,
69 const SearchContext& context)
70{
71 const auto state_trajectory = extract_state_trajectory(search_nodes, final_search_node, final_state_index);
72
73 auto actions = formalism::GroundActionList {};
74 auto states = StateList { start_state };
75
76 auto state = start_state;
77 auto state_metric_value = start_state_metric_value;
78
79 for (size_t i = 0; i < state_trajectory.size() - 1; ++i)
80 {
81 // We have to take the (state,action) pair that yields lowest metric value.
82 auto lowest_action = formalism::GroundAction { nullptr };
83 auto lowest_state = std::optional<State> { std::nullopt };
84 auto lowest_metric_value = std::numeric_limits<ContinuousCost>::infinity();
85
86 for (const auto& action : context->get_applicable_action_generator()->create_applicable_action_generator(state))
87 {
88 const auto [successor_state, successor_state_metric_value] =
89 context->get_state_repository()->get_or_create_successor_state(state, action, state_metric_value);
90 if (successor_state.get_index() == state_trajectory.at(i + 1))
91 {
92 if (successor_state_metric_value < lowest_metric_value)
93 {
94 lowest_action = action;
95 lowest_state = successor_state;
96 lowest_metric_value = successor_state_metric_value;
97 }
98 }
99 }
100 assert(lowest_state && lowest_action);
101 actions.push_back(lowest_action);
102 states.push_back(lowest_state.value());
103 state = lowest_state.value();
104 state_metric_value = lowest_metric_value;
105 }
106
107 return Plan(context, std::move(states), std::move(actions), state_metric_value);
108}
