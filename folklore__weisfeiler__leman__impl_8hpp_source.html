79template<size_t K>
80size_t tuple_to_hash(const IndexArray<K>& tuple, size_t num_vertices)
81{
82 size_t hash = 0;
83 size_t weight = 1;
84 for (size_t i = 0; i < K; ++i)
85 {
86 hash += weight * tuple[i];
87 weight *= num_vertices;
88 }
89 return hash;
90}

92template<size_t K>
93IndexArray<K> hash_to_tuple(size_t hash, size_t num_vertices)
94{
95 auto result = IndexArray<K>();
96 auto weight = (size_t) std::pow(num_vertices, K - 1);
97 for (int64_t i = K - 1; i >= 0; --i)
98 {
99 result[i] = hash / weight;
100 hash %= weight;
101 weight /= num_vertices;
102 }
103 return result;
104}

112template<size_t K, typename G>
113 requires IsVertexListGraph<G> && IsIncidenceGraph<G> && IsVertexColoredGraph<G> //
114std::pair<ColorList, ColorMap<IndexList>> compute_ordered_isomorphism_types(const G& graph, IsomorphismTypeCompressionFunction& iso_type_function)
115{
116 const auto num_vertices = graph.get_num_vertices();
117 const auto num_hashes = std::pow(num_vertices, K);
118
119 /* Temporary bookkeeping to support dynamic graphs. */
120 auto vertex_to_v = IndexMap<Index>();
121 auto v_to_vertex = IndexMap<Index>();
122 for (const auto& vertex : graph.get_vertex_indices())
123 {
124 const auto v = Index(vertex_to_v.size());
125 vertex_to_v.emplace(vertex, v);
126 v_to_vertex.emplace(v, vertex);
127 }
128
129 // Create adj matrix for fast creation of subgraph induced by k-tuple.
130 auto adj_matrix = std::vector<std::vector<bool>>(num_vertices, std::vector<bool>(num_vertices, false));
131 for (const auto& vertex1 : graph.get_vertex_indices())
132 {
133 for (const auto& vertex2 : graph.template get_adjacent_vertex_indices<ForwardTag>(vertex1))
134 {
135 adj_matrix.at(vertex_to_v.at(vertex1)).at(vertex_to_v.at(vertex2)) = true;
136 }
137 }
138
139 auto hash_to_color = ColorList(num_hashes);
140 auto color_to_hashes = ColorMap<IndexList>();
141
142 auto subgraph = nauty_wrapper::SparseGraph(K);
143 auto subgraph_coloring = ColorList();
144
145 // Subroutine to compute (ordered) isomorphic types of all k-tuples of vertices.
146 auto v_to_i = IndexMap<Index>();
147 for (size_t hash = 0; hash < num_hashes; ++hash)
148 {
149 // Compress indexing of subgraph.
150 v_to_i.clear();
151 auto t = hash_to_tuple<K>(hash, num_vertices);
152 for (size_t i = 0; i < K; ++i)
153 {
154 v_to_i.emplace(t[i], v_to_i.size());
155 }
156
157 // Initialize empty subgraph and coloring
158 const auto subgraph_num_vertices = v_to_i.size();
159 subgraph.clear(subgraph_num_vertices);
160 subgraph_coloring.resize(subgraph_num_vertices);
161
162 // Instantiate vertex-colored subgraph.
163 for (const auto [v1, i1] : v_to_i)
164 {
165 for (const auto [v2, i2] : v_to_i)
166 {
167 if (adj_matrix[v1][v2])
168 {
169 subgraph.add_edge(i1, i2);
170 }
171 }
172 subgraph_coloring[i1] = get_color(graph.get_vertex(v_to_vertex.at(v1)));
173 }
174 subgraph.add_vertex_coloring(subgraph_coloring);
175
176 // Isomorphism function is shared among several runs to ensure canonical form for different runs.
177 auto result = iso_type_function.emplace(subgraph.compute_certificate(), iso_type_function.size());
178
179 const auto color = result.first->second;
180 hash_to_color.at(hash) = color;
181 color_to_hashes[color].push_back(hash);
182 }
183
184 return std::make_pair(hash_to_color, color_to_hashes);
185}

187template<size_t K, typename G>
188 requires IsVertexListGraph<G> && IsIncidenceGraph<G> && IsVertexColoredGraph<G> //
189Certificate<K> compute_certificate(const G& graph, IsomorphismTypeCompressionFunction& iso_type_function)
190{
191 if (!is_undirected_graph(graph))
192 {
193 throw std::runtime_error("K-FWL does not support directed graphs because they can be translated into undirected graphs by introducing two vertices "
194 "along the edge with different colors to encode the direction.");
195 }
196
197 // Toggle verbosity
198 const bool debug = false;
199
200 /* Fetch some data. */
201 const auto num_vertices = graph.get_num_vertices();
202 const auto num_hashes = std::pow(num_vertices, K);
203
204 if (debug)
205 {
206 std::cout << "num_vertices=" << num_vertices << std::endl //
207 << "num_hashes=" << num_hashes << std::endl;
208 }
209
210 /* Compute max color used in graph. */
211 auto max_color = Color();
212 for (const auto& vertex : graph.get_vertex_indices())
213 {
214 max_color = std::max(max_color, get_color(graph.get_vertex(vertex)));
215 }
216
217 /* Compute isomorphism types. */
218 auto hash_to_color = ColorList();
219 auto color_to_hashes = ColorMap<IndexList>();
220
221 const auto [hash_to_color_, color_to_hashes_] = compute_ordered_isomorphism_types<K>(graph, iso_type_function);
222
223 hash_to_color = std::move(hash_to_color_);
224 color_to_hashes = std::move(color_to_hashes_);
225
226 /* Initialize work list of color. */
227 auto L = ColorSet(hash_to_color.begin(), hash_to_color.end());
228
229 /* Refine colors of k-tuples. */
230 auto f = typename Certificate<K>::ConfigurationCompressionFunction();
231 auto M = std::vector<std::pair<Index, ColorArray<K>>>();
232 auto M_replaced = std::vector<std::tuple<Color, std::vector<ColorArray<K>>, Index>>();
233 // (line 3-18): subroutine to find stable coloring
234
235 while (!L.empty())
236 {
237 if (debug)
238 std::cout << "L: " << L << std::endl;
239
240 // Clear data structures that are reused.
241 M.clear();
242 M_replaced.clear();
243
244 {
245 // (lines 4-14): Subroutine to fill multiset
246 // Note: this computes the stable coloring, not the coarsest stable coloring.
247 for (size_t h = 0; h < num_hashes; ++h)
248 {
249 const auto w = hash_to_tuple<K>(h, num_vertices);
250
251 for (size_t j = 0; j < K; ++j)
252 {
253 for (size_t u = 0; u < num_vertices; ++u)
254 {
255 // C[\vec{v}[1,u]],...,C[\vec{v}[k,u]]
256 auto k_coloring = ColorArray<K>();
257 for (size_t i = 0; i < K; ++i)
258 {
259 // \vec{x} = \vec{v}[i,u]
260 auto x = w;
261 x[i] = u;
262 const auto x_hash = tuple_to_hash<K>(x, num_vertices);
263
264 k_coloring.at(i) = hash_to_color.at(x_hash);
265 }
266
267 M.emplace_back(h, std::move(k_coloring));
268 }
269 }
270 }
271 }
272
273 // (line 15): Perform radix sort of M
274 std::sort(M.begin(), M.end());
275
276 if (debug)
277 std::cout << "M: " << M << std::endl;
278
279 // (line 16): Scan M and replace tuples (vec{v},c_1^1,...,c_k^1,...,vec{v},c_1^r,...,c_k^r) with single tuple
280 // (C(vec{v}),(c_1^1,...,c_k^1),...,(c_1^r,...,c_k^r))
281 color_refinement::replace_tuples(M, hash_to_color, M_replaced);
282
283 // (line 17): Perform radix sort of M
284 std::sort(M_replaced.begin(), M_replaced.end());
285
286 if (debug)
287 std::cout << "M_replaced: " << M_replaced << std::endl;
288
289 // (line 18): Split color classes
290 color_refinement::split_color_classes(M_replaced, f, max_color, hash_to_color, color_to_hashes, L);
291 }
292
293 /* Report final neighborhood structures in the decoding table. */
294 for (const auto& [old_color, signature, hash] : M_replaced)
295 {
296 f.emplace(std::make_pair(old_color, signature), old_color);
297 }
298
299 /* Return the certificate */
300 return Certificate(std::move(f), std::move(hash_to_color));
301}
