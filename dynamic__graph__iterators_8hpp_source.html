329template<IsVertex Vertex, IsEdge Edge, IsDirection Direction>
330DynamicAdjacentVertexConstIterator<Vertex, Edge, Direction>::reference DynamicAdjacentVertexConstIterator<Vertex, Edge, Direction>::operator*() const
331{
332 assert(m_vertices);
333 assert(m_edges);
334
335 if constexpr (std::is_same_v<Direction, ForwardTag>)
336 {
337 return m_vertices->at(m_edges->at(*m_slice_iter).get_target());
338 }
339 else if constexpr (std::is_same_v<Direction, BackwardTag>)
340 {
341 return m_vertices->at(m_edges->at(*m_slice_iter).get_source());
342 }
343 else
344 {
345 static_assert(dependent_false<Direction>::value,
346 "DynamicAdjacentVertexConstIterator<Edge, Direction>::operator*(): Missing implementation for IsDirection.");
347 }
348}
