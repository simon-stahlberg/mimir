36class GeneratedSentencesContainer
37{
38private:
39 template<FeatureCategory D>
40 using GeneratedConstructorsMap = std::unordered_map<NonTerminal<D>, std::vector<dl::ConstructorList<D>>>;
41
42 template<FeatureCategory... Ds>
43 using HanaGeneratedConstructorsMaps = boost::hana::map<boost::hana::pair<boost::hana::type<Ds>, GeneratedConstructorsMap<Ds>>...>;
44
45 static const dl::ConstructorLists empty_lists;
46
47 HanaGeneratedConstructorsMaps<Concept, Role, Boolean, Numerical> m_generated_constructors;
48
49public:
50 template<FeatureCategory D>
51 dl::ConstructorList<D>& get(NonTerminal<D> nonterminal, size_t complexity)
52 {
53 auto& constructors_by_complexity = boost::hana::at_key(m_generated_constructors, boost::hana::type<D> {})[nonterminal];
54 if (complexity >= constructors_by_complexity.size())
55 {
56 constructors_by_complexity.resize(complexity + 1);
57 }
58 return constructors_by_complexity[complexity];
59 }
60
61 template<FeatureCategory D>
62 const dl::ConstructorList<D>& get(NonTerminal<D> nonterminal, size_t complexity) const
63 {
64 const auto& container = boost::hana::at_key(m_generated_constructors, boost::hana::type<D> {});
65
66 auto it = container.find(nonterminal);
67 if (it == container.end() || complexity >= it->second.size())
68 {
69 return boost::hana::at_key(empty_lists, boost::hana::type<D> {});
70 }
71
72 return it->second.at(complexity);
73 }
74
75 HanaGeneratedConstructorsMaps<Concept, Role, Boolean, Numerical>& get_hana_generated_constructors() { return m_generated_constructors; }
76};
