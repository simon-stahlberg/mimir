46public:
47 Denotation<D> evaluate(EvaluationContext& context) const override
48 {
49 // Try to access cached result
50 auto denotation = boost::hana::at_key(context.get_repositories(), boost::hana::type<D> {}).get_if(this, context.get_state());
51
52 if (denotation)
53 {
54 return denotation;
55 }
56
57 // Compute the result in the derived class
58 self().evaluate_impl(context);
59
60 // Store and return result;

62 .insert(this, context.get_state(), boost::hana::at_key(context.get_builders(), boost::hana::type<D> {}));
63 };
64
65 void accept(IVisitor& visitor) const override { return self().accept_impl(visitor); }
