136template<std::ranges::forward_range VertexIndices>
137 requires std::same_as<std::ranges::range_value_t<VertexIndices>, graphs::VertexIndex>
138GeneralPolicyImpl::UnsolvabilityReason
139GeneralPolicyImpl::solves(const datasets::StateSpace& state_space, const VertexIndices& vertices, dl::DenotationRepositories& denotation_repositories) const
140{
141 const auto& graph = state_space->get_graph();
142
143 using IteratorType = graphs::ProblemGraph::AdjacentVertexIndexConstIteratorType<graphs::Forward>;
144
145 struct Entry
146 {
147 graphs::VertexIndex v_idx;
148 IteratorType it;
149 IteratorType end;
150 bool has_compatible_edge;
151 };
152
153 auto visited_v_idxs = graphs::VertexIndexSet {};
154 auto stack_v_idxs = graphs::VertexIndexSet {};
155
156 DEBUG_LOG(graph);
157
158 for (const auto& v_idx : vertices)
159 {
160 if (visited_v_idxs.contains(v_idx))
161 continue;
162
163 auto stack = std::stack<Entry> {};
164 stack.push(Entry { v_idx,
165 graph.get_adjacent_vertex_indices<graphs::Forward>(v_idx).begin(),
166 graph.get_adjacent_vertex_indices<graphs::Forward>(v_idx).end(),
167 false });
168 visited_v_idxs.insert(v_idx);
169 stack_v_idxs.insert(v_idx);
170
171 while (!stack.empty())
172 {
173 auto& src_entry = stack.top();
174 const auto src_v_idx = src_entry.v_idx;
175 const auto& src_v = graph.get_vertex(src_v_idx);
176 const auto& src_problem = graphs::get_problem(src_v);
177 const auto src_state = graphs::get_state(src_v);
178
179 if (src_entry.it == src_entry.end)
180 {
181 /* No more dst_v_idx available. */
182
183 if (!src_entry.has_compatible_edge && !graphs::is_goal(src_v))
184 {
185 DEBUG_LOG("\nUnsolvable vertex " << src_v_idx << ": " << std::make_tuple(src_state, std::cref(*src_problem)));
186
187 return UnsolvabilityReason::UNSOLVABLE;
188 }
189
190 stack.pop();
191 stack_v_idxs.erase(src_entry.v_idx);
192 }
193 else
194 {
195 auto src_eval_context = dl::EvaluationContext(src_state, src_problem, denotation_repositories);
196
197 const auto dst_v_idx = *src_entry.it++;
198 const auto& dst_v = graph.get_vertex(dst_v_idx);
199 const auto& dst_problem = graphs::get_problem(dst_v);
200 const auto dst_state = graphs::get_state(dst_v);
201 auto dst_eval_context = dl::EvaluationContext(dst_state, dst_problem, denotation_repositories);
202
203 const bool is_compatible = evaluate(src_eval_context, dst_eval_context);
204
205 if (is_compatible)
206 {
207 if (stack_v_idxs.contains(dst_v_idx))
208 {
209 DEBUG_LOG("\nCompatible cyclic edge " << src_v_idx << "->" << dst_v_idx << ":\n"
210 << std::make_tuple(src_state, std::cref(*src_problem)) << "\n"
211 << " -> " << std::make_tuple(dst_state, std::cref(*dst_problem)));
212
213 return UnsolvabilityReason::CYCLE;
214 }
215
216 src_entry.has_compatible_edge = true;
217
218 if (!visited_v_idxs.contains(dst_v_idx))
219 {
220 DEBUG_LOG("\nCompatible edge " << src_v_idx << "->" << dst_v_idx << ":\n"
221 << std::make_tuple(src_state, std::cref(*src_problem)) << "\n"
222 << " -> " << std::make_tuple(dst_state, std::cref(*dst_problem)));
223
224 stack.push(Entry { dst_v_idx,
225 graph.get_adjacent_vertex_indices<graphs::Forward>(dst_v_idx).begin(),
226 graph.get_adjacent_vertex_indices<graphs::Forward>(dst_v_idx).end(),
227 false });
228 visited_v_idxs.insert(dst_v_idx);
229 stack_v_idxs.insert(dst_v_idx);
230 }
231 }
232 else
233 {
234 DEBUG_LOG("\nIncompatible edge " << src_v_idx << "->" << dst_v_idx << ":\n"
235 << std::make_tuple(src_state, std::cref(*src_problem)) << "\n"
236 << " -> " << std::make_tuple(dst_state, std::cref(*dst_problem)));
237 }
238 }
239 }
240 }
241
242 return UnsolvabilityReason::NONE;
243}
