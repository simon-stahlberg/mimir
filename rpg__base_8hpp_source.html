287 std::tuple<IsPreconditionOfContainer, IndexList> instantiate_actions(const DeleteRelaxedProblemExplorator& delete_relaxation)
288 {
289 auto& actions = get_structures<Action>();
290 auto is_precondition_of_action = IsPreconditionOfContainer {};
291 auto trivial_unary_actions = IndexList {};
292
293 for (const auto& action : delete_relaxation.create_ground_actions())
294 {
295 for (const auto& cond_effect : action->get_conditional_effects())
296 {
297 const auto num_preconditions = action->get_conjunctive_condition()->get_num_preconditions<formalism::FluentTag, formalism::DerivedTag>()
298 + cond_effect->get_conjunctive_condition()->get_num_preconditions<formalism::FluentTag, formalism::DerivedTag>();
299
300 auto preconditions = action->get_conjunctive_condition()->get_hana_compressed_precondition<formalism::FluentTag, formalism::DerivedTag>();
301 auto conditional_preconditions =
302 cond_effect->get_conjunctive_condition()->get_hana_compressed_precondition<formalism::FluentTag, formalism::DerivedTag>();
303
304 for (const auto& eff_atom_index : cond_effect->get_conjunctive_effect()->get_propositional_effects<formalism::PositiveTag>())
305 {
306 const auto unary_action_index = actions.size();
307
308 /* Positive */
309 for (const auto& pre_atom_index : action->get_conjunctive_condition()->get_precondition<formalism::PositiveTag, formalism::FluentTag>())
310 {
311 get<formalism::PositiveTag, formalism::FluentTag>(is_precondition_of_action)[pre_atom_index].push_back(unary_action_index);
312 }
313 for (const auto& pre_atom_index : action->get_conjunctive_condition()->get_precondition<formalism::PositiveTag, formalism::DerivedTag>())
314 {
315 get<formalism::PositiveTag, formalism::DerivedTag>(is_precondition_of_action)[pre_atom_index].push_back(unary_action_index);
316 }
317 for (const auto& pre_atom_index :
318 cond_effect->get_conjunctive_condition()->get_precondition<formalism::PositiveTag, formalism::FluentTag>())
319 {
320 get<formalism::PositiveTag, formalism::FluentTag>(is_precondition_of_action)[pre_atom_index].push_back(unary_action_index);
321 }
322 for (const auto& pre_atom_index :
323 cond_effect->get_conjunctive_condition()->get_precondition<formalism::PositiveTag, formalism::DerivedTag>())
324 {
325 get<formalism::PositiveTag, formalism::DerivedTag>(is_precondition_of_action)[pre_atom_index].push_back(unary_action_index);
326 }
327 /* Negative */
328 for (const auto& pre_atom_index : action->get_conjunctive_condition()->get_precondition<formalism::NegativeTag, formalism::FluentTag>())
329 {
330 get<formalism::NegativeTag, formalism::FluentTag>(is_precondition_of_action)[pre_atom_index].push_back(unary_action_index);
331 }
332 for (const auto& pre_atom_index : action->get_conjunctive_condition()->get_precondition<formalism::NegativeTag, formalism::DerivedTag>())

334 get<formalism::NegativeTag, formalism::DerivedTag>(is_precondition_of_action)[pre_atom_index].push_back(unary_action_index);

344 get<formalism::NegativeTag, formalism::DerivedTag>(is_precondition_of_action)[pre_atom_index].push_back(unary_action_index);

347 actions.push_back(Action(unary_action_index, action, preconditions, conditional_preconditions, num_preconditions, eff_atom_index, true));
348
349 if (actions.back().get_num_preconditions() == 0)
350 {
351 trivial_unary_actions.push_back(unary_action_index);

353 }
354 for (const auto& eff_atom_index : cond_effect->get_conjunctive_effect()->get_propositional_effects<formalism::NegativeTag>())
355 {
356 const auto unary_action_index = actions.size();
357
358 /* Positive */
359 for (const auto& pre_atom_index : action->get_conjunctive_condition()->get_precondition<formalism::PositiveTag, formalism::FluentTag>())

422 {
423 get<formalism::PositiveTag, formalism::FluentTag>(is_precondition_of_axiom)[pre_atom_index].push_back(unary_axiom_index);
424 }
425 for (const auto& pre_atom_index : axiom->get_conjunctive_condition()->get_precondition<formalism::PositiveTag, formalism::DerivedTag>())
426 {
427 get<formalism::PositiveTag, formalism::DerivedTag>(is_precondition_of_axiom)[pre_atom_index].push_back(unary_axiom_index);
428 }
429 /* Negative */
430 for (const auto& pre_atom_index : axiom->get_conjunctive_condition()->get_precondition<formalism::NegativeTag, formalism::FluentTag>())
431 {
432 get<formalism::NegativeTag, formalism::FluentTag>(is_precondition_of_axiom)[pre_atom_index].push_back(unary_axiom_index);
433 }
434 for (const auto& pre_atom_index : axiom->get_conjunctive_condition()->get_precondition<formalism::NegativeTag, formalism::DerivedTag>())

436 get<formalism::NegativeTag, formalism::DerivedTag>(is_precondition_of_axiom)[pre_atom_index].push_back(unary_axiom_index);
437 }

465 template<formalism::IsPolarity R, formalism::IsFluentOrDerivedTag P>
466 void instantiate_propositions(const formalism::ProblemImpl& problem,
467 IsPreconditionOfContainer& is_precondition_of_action,
468 IsPreconditionOfContainer& is_precondition_of_axiom)
469 {
470 const auto& ground_atom_repository =
471 boost::hana::at_key(problem.get_repositories().get_hana_repositories(), boost::hana::type<formalism::GroundAtomImpl<P>> {});
472
473 auto& offsets = get_offsets<R, P>();
474
475 for (Index atom_index = 0; atom_index < ground_atom_repository.size(); ++atom_index)
476 {
477 const auto proposition_index = m_propositions.size();
478
479 if (atom_index >= offsets.size())
480 {
481 offsets.resize(atom_index + 1, MAX_INDEX);
482 }
483 offsets[atom_index] = proposition_index;
484
485 const auto is_goal = problem.template get_goal_atoms_bitset<R, P>().get(atom_index);
486
487 m_propositions.emplace_back(
488 proposition_index,
489 boost::hana::make_map(boost::hana::make_pair(boost::hana::type<Action> {}, std::move(get<R, P>(is_precondition_of_action)[atom_index])),
490 boost::hana::make_pair(boost::hana::type<Axiom> {}, std::move(get<R, P>(is_precondition_of_axiom)[atom_index]))),
491 is_goal);
492
493 if (is_goal)
494 m_goal_propositions.push_back(proposition_index);
495 }

519 template<formalism::IsFluentOrDerivedTag P>
520 void initialize_or_annotations_and_queue_helper(State state)
521 {
522 if constexpr (std::is_same_v<P, formalism::DerivedTag>)
523 {
524 // Derive all negative derived propositions immediately, i.e., not y <- T for all y.
525
526 auto& positive_offsets = get_offsets<formalism::PositiveTag, P>();
527 auto& negative_offsets = get_offsets<formalism::NegativeTag, P>();
528
529 for (const auto& atom_index : state->get_atoms<P>())
530 {
531 self().initialize_or_annotations_and_queue_impl(m_propositions[positive_offsets[atom_index]]);
532 }
533
534 for (const auto& atom_index : get_atom_indices<P>())
535 {
536 self().initialize_or_annotations_and_queue_impl(m_propositions[negative_offsets[atom_index]]);
537 }
538 }
539 else
540 {
541 auto& positive_offsets = get_offsets<formalism::PositiveTag, P>();
542 auto& negative_offsets = get_offsets<formalism::NegativeTag, P>();
543 const auto& all_atoms = get_atom_indices<P>();
544 const auto& state_atoms = state->get_atoms<P>();
545 auto it = state_atoms.begin();
546 auto it2 = all_atoms.begin();
547 const auto end = state_atoms.end();
548 const auto end2 = all_atoms.end();
549
550 while (it != end && it2 != end2)
551 {
552 if (*it == *it2)
553 {
554 self().initialize_or_annotations_and_queue_impl(m_propositions[positive_offsets[*it]]);
555 ++it;
556 ++it2;
557 }
558 else if (*it < *it2)
559 {
560 self().initialize_or_annotations_and_queue_impl(m_propositions[positive_offsets[*it]]);
561 ++it;
562 }
563 else
564 {
565 self().initialize_or_annotations_and_queue_impl(m_propositions[negative_offsets[*it2]]);
566 ++it2;
567 }
568 }
569 while (it != end)
570 {
571 self().initialize_or_annotations_and_queue_impl(m_propositions[positive_offsets[*it]]);
572 ++it;
573 }
574 while (it2 != end2)
575 {
576 self().initialize_or_annotations_and_queue_impl(m_propositions[negative_offsets[*it2]]);
577 ++it2;
578 }
579 }
580 }

593 template<IsStructure S>
594 void on_process_structure(const Proposition& proposition, const S& structure)
595 {
596 auto& structure_annotation = get_structures_annotations<S>()[structure.get_index()];
597
598 self().update_and_annotation_impl(proposition, structure);
599
600 if (proposition.get_index() != DUMMY_PROPOSITION_INDEX)
601 {
602 --get_num_unsatisfied_preconditions(structure_annotation);
603 }
604
605 if (get_num_unsatisfied_preconditions(structure_annotation) == 0)
606 {
607 // std::cout << "Process satisfied structure: " << structure.get_index() << std::endl;
608
609 const auto effect_proposition_index = structure.get_polarity() ?
610 get_offsets<formalism::PositiveTag, formalism::FluentTag>()[structure.get_effect()] :
611 get_offsets<formalism::NegativeTag, formalism::FluentTag>()[structure.get_effect()];
612
613 const auto& effect_proposition = get_propositions()[effect_proposition_index];
614
615 self().update_or_annotation_impl(structure, effect_proposition);
616 }

619 void dijksta()
620 {
621 m_num_unsat_goals = m_goal_propositions.size();
622
623 while (!m_queue.empty())
624 {
