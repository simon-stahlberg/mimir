177 self().on_finish_f_layer_impl(f_value,

205 void on_end_search(uint64_t num_reached_fluent_atoms,

213

217 m_statistics.set_num_reached_derived_atoms(num_reached_derived_atoms);

219 m_statistics.set_num_bytes_for_nodes(num_bytes_for_nodes);

221 m_statistics.set_num_nodes(num_nodes);
222 m_statistics.set_num_actions(num_actions);
223 m_statistics.set_num_axioms(num_axioms);
224
225 if (!m_quiet)
226 {
227 self().on_end_search_impl(num_reached_fluent_atoms,
228 num_reached_derived_atoms,
229 num_bytes_for_problem,
230 num_bytes_for_nodes,
231 num_states,
232 num_nodes,
233 num_actions,
234 num_axioms);
235 }

242 self().on_solved_impl(plan);

250 self().on_unsolvable_impl();
251 }
252 }
253

255 {

258 self().on_exhausted_impl();
259 }
