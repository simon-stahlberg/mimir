102 template<FeatureCategory D>
103 const DerivationRuleList<D>& get(NonTerminal<D> non_terminal) const
104 {
105 const auto& container = boost::hana::at_key(m_map, boost::hana::type<D> {});

107 auto it = container.find(non_terminal);
108 if (it == container.end())
