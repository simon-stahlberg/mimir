78template<IsArithmetic A>
79inline bool evaluate(loki::BinaryComparatorEnum comparator, const Bounds<A>& lhs, const Bounds<A>& rhs)
80{
81 switch (comparator)
82 {
83 case loki::BinaryComparatorEnum::EQUAL:
84 {
85 return lhs == rhs;
86 }
87 case loki::BinaryComparatorEnum::GREATER:
88 {
89 return lhs > rhs;
90 }
91 case loki::BinaryComparatorEnum::GREATER_EQUAL:
92 {
93 return lhs >= rhs;
94 }
95 case loki::BinaryComparatorEnum::LESS:
96 {
97 return lhs < rhs;
98 }
99 case loki::BinaryComparatorEnum::LESS_EQUAL:
100 {
101 return lhs <= rhs;
102 }
103 default:
104 {
105 throw std::logic_error("evaluate(comparator, lhs, rhs): Unexpected BinaryComparatorEnum.");
106 }
107 }
108}
