152inline std::ostream& operator<<(std::ostream& os, const Statistics& statistics)
153{
154 os << "[AStar] Search time: " << statistics.get_search_time_ms().count() << "ms" << "\n"
155 << "[AStar] Number of generated states: " << statistics.get_num_generated() << "\n"
156 << "[AStar] Number of expanded states: " << statistics.get_num_expanded() << "\n"
157 << "[AStar] Number of pruned states: " << statistics.get_num_pruned() << "\n"
158 << "[AStar] Number of generated states until last f-layer: "
159 << (statistics.get_num_generated_until_f_value().empty() ? 0 : statistics.get_num_generated_until_f_value().rbegin()->second) << "\n"
160 << "[AStar] Number of expanded states until last f-layer: "
161 << (statistics.get_num_expanded_until_f_value().empty() ? 0 : statistics.get_num_expanded_until_f_value().rbegin()->second) << "\n"
162 << "[AStar] Number of pruned states until last f-layer: "
163 << (statistics.get_num_pruned_until_f_value().empty() ? 0 : statistics.get_num_pruned_until_f_value().rbegin()->second) << "\n"
164 << "[AStar] Number of reached fluent atoms: " << statistics.get_num_reached_fluent_atoms() << "\n"
165 << "[AStar] Number of reached derived atoms: " << statistics.get_num_reached_derived_atoms() << "\n"
166 << "[AStar] Number of states: " << statistics.get_num_states() << "\n"
167 << "[AStar] Number of nodes: " << statistics.get_num_nodes() << "\n"
168 << "[AStar] Number of bytes for nodes: " << statistics.get_num_bytes_for_nodes() << "\n"
169 << "[AStar] Number of bytes per node: " << divide_or_zero(statistics.get_num_bytes_for_nodes(), statistics.get_num_nodes()) << "\n"
170 << "[AStar] Number of bytes for problem: " << statistics.get_num_bytes_for_problem() << "\n"
171 << "[AStar] Total number of bytes used: " << (statistics.get_num_bytes_for_problem() + statistics.get_num_bytes_for_nodes());
