124template<typename T>
125class UniqueObjectPool
126{
127private:
128 std::vector<std::unique_ptr<T>> m_storage;
129 std::stack<T*> m_stack;
130
131 template<typename... Args>
132 void allocate(Args&&... args)
133 {
134 m_storage.push_back(std::make_unique<T>(T(std::forward<Args>(args)...)));
135 m_stack.push(m_storage.back().get());
136 }
137
138 void free(T* element) { m_stack.push(element); }
139
140 friend class UniqueObjectPoolPtr<T>;
141
142public:
143 // Non-copyable to prevent dangling memory pool pointers.
144 UniqueObjectPool() = default;
145 UniqueObjectPool(const UniqueObjectPool& other) = delete;
146 UniqueObjectPool& operator=(const UniqueObjectPool& other) = delete;
147 UniqueObjectPool(UniqueObjectPool&& other) = delete;
148 UniqueObjectPool& operator=(UniqueObjectPool&& other) = delete;
149
150 [[nodiscard]] UniqueObjectPoolPtr<T> get_or_allocate() { return get_or_allocate<>(); }
151
152 template<typename... Args>
153 [[nodiscard]] UniqueObjectPoolPtr<T> get_or_allocate(Args&&... args)
154 {
155 if (m_stack.empty())
156 {
157 allocate(std::forward<Args>(args)...);
158 }
159 T* element = m_stack.top();
160 m_stack.pop();
161 return UniqueObjectPoolPtr<T>(this, element);
162 }
163
164 [[nodiscard]] size_t get_size() const { return m_storage.size(); }
165
166 [[nodiscard]] size_t get_num_free() const { return m_stack.size(); }
167};
