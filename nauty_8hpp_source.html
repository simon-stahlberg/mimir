70class DenseGraph
71{
72private:
73 std::unique_ptr<DenseGraphImpl> m_impl;
74
75public:
77 DenseGraph();
78
81 explicit DenseGraph(size_t num_vertices);
82
83 template<mimir::graphs::IsVertexColoredGraph Graph>
84 explicit DenseGraph(const Graph& graph) : DenseGraph(graph.get_num_vertices())
85 {
86 /* Remap indices to 0,1,... indexing schema. */
87 auto remap = mimir::IndexMap<mimir::Index>();
88 for (const auto& vertex : graph.get_vertices())
89 {
90 remap.emplace(vertex.get_index(), remap.size());
91 }
92
93 /* Add edges. */
94 for (const auto& edge : graph.get_edges())
95 {
96 add_edge(remap.at(edge.get_source()), remap.at(edge.get_target()));
97 }
98
99 /* Add vertex coloring. */
100 auto coloring = mimir::graphs::ColorList(graph.get_num_vertices());
101 for (const auto& vertex : graph.get_vertices())
102 {
103 coloring.at(remap.at(vertex.get_index())) = get_color(vertex);
104 }
105 add_vertex_coloring(coloring);
106 }
107
108 DenseGraph(const DenseGraph& other);
109 DenseGraph& operator=(const DenseGraph& other);
110 DenseGraph(DenseGraph&& other) noexcept;
111 DenseGraph& operator=(DenseGraph&& other) noexcept;
112 ~DenseGraph();
113
119 void add_vertex_coloring(const mimir::graphs::ColorList& vertex_coloring);
120
126 void add_edge(size_t source, size_t target);
127
130 void clear(size_t num_vertices);
131
133 bool is_directed() const;
134
135 friend Certificate compute_certificate(const DenseGraph& graph);
