79 template<typename Graph>
80 requires IsVertexListGraph<Graph> && IsEdgeListGraph<Graph>
81 explicit SparseGraph(const Graph& graph) : SparseGraph()
82 {
83 if (is_multi(graph))
84 {
85 throw std::runtime_error("SparseGraph::SparseGraph(graph): Multi graph is not supported.");
86 }
87 if (!is_loopless(graph))
88 {
89 throw std::runtime_error("SparseGraph::SparseGraph(graph): looping edges are not supported.");
90 }
91 if (!is_undirected(graph))
92 {
93 throw std::runtime_error("SparseGraph::SparseGraph(graph): directed graphs are not supported due to bad performance. Please translate it into an "
94 "undirected graph before using nauty.");
95 }
96
97 /* Remap indices to 0,1,... indexing schema. */
98 auto remap = mimir::IndexMap<uint32_t>();
99 for (const auto& vertex : graph.get_vertices())
100 {
101 remap.emplace(vertex.get_index(), remap.size());
102 }
103
104 // Copies from nauty sparsegraph and wrapped into c++ types.
105 size_t nde; /* Number of directed edges (loops contribute only 1) */
106 std::vector<size_t> v; /* Array of indexes into e[*] */
107 int nv; /* Number of vertices */
108 std::vector<int> d; /* Array with out-degree of each vertex */
109 std::vector<int> e; /* Array to hold lists of neighbours */
110 size_t vlen, dlen, elen; /* Sizes of arrays in units of type */
111
112 nv = graph.get_num_vertices(); /* Number of directed edges (loops contribute only 1) */
113 nde = graph.get_num_edges(); /* Number of vertices */
114 vlen = nv;
115 dlen = nv;
116 elen = nde;
117
118 d.reserve(nv);
119 v.reserve(nv);
120 e.reserve(nde);
121 for (const auto& vertex : graph.get_vertices())
122 {
123 d.push_back(graph.template get_degree<ForwardTag>(vertex.get_index()));
124 v.push_back(e.size());
125 for (const auto& edge : graph.template get_adjacent_edges<ForwardTag>(vertex.get_index()))
126 {
127 e.push_back(remap.at(edge.get_target()));
128 }
129 }
130
131 /* Add vertex coloring. */
132
133 auto color_vertex_pairs = std::vector<std::pair<PropertyValue, uint32_t>> {};
134 color_vertex_pairs.reserve(nv);
135 for (const auto& vertex : graph.get_vertices())
136 {
137 color_vertex_pairs.emplace_back(vertex.get_properties(), remap.at(vertex.get_index()));
138 }
139 std::sort(color_vertex_pairs.begin(), color_vertex_pairs.end());
140
141 auto coloring = PropertyValueList {};
142 coloring.reserve(nv);
143 for (const auto& [color, _] : color_vertex_pairs)
144 {
145 coloring.push_back(color);
146 }
147
148 auto lab = std::vector<int>(nv, 0);
149 auto ptn = std::vector<int>(nv, 0);
150 for (int i = 1; i < nv; ++i)
151 {
152 lab[i - 1] = color_vertex_pairs[i - 1].second;
153 ptn[i - 1] = (color_vertex_pairs[i - 1].first != color_vertex_pairs[i].first) ? 0 : 1;
154 }
155 if (nv > 0)
156 {
157 lab[nv - 1] = color_vertex_pairs[nv - 1].second;
158 ptn[nv - 1] = 0;
159 }
160
161 initialize(nde, v, nv, d, e, vlen, dlen, elen, lab, ptn, coloring);
162 }
