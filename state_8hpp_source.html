83template<formalism::IsFluentOrDerivedTag P>
84auto StateImpl::get_atoms() const
85{
86 if constexpr (std::is_same_v<P, formalism::FluentTag>)
87 {
88 return m_fluent_atoms->compressed_range();
89 }
90 else if constexpr (std::is_same_v<P, formalism::DerivedTag>)
91 {
92 return m_derived_atoms->compressed_range();
93 }
94 else
95 {
96 static_assert(dependent_false<P>::value, "Missing implementation for IsStaticOrFluentOrDerivedTag.");
97 }
98}

100template<formalism::IsFluentOrDerivedTag P, std::ranges::forward_range Range1, std::ranges::forward_range Range2>
101 requires IsRangeOver<Range1, Index> && IsRangeOver<Range2, Index>
102bool StateImpl::literals_hold(const Range1& positive_atoms, const Range2& negative_atoms) const
103{
104 return is_supseteq(get_atoms<P>(), positive_atoms) && are_disjoint(get_atoms<P>(), negative_atoms);
