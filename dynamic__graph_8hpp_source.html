221template<IsVertex V, IsEdge E>
222template<typename... VertexProperties>
223 requires HasVertexProperties<V, VertexProperties...>
224VertexIndex DynamicGraph<V, E>::add_vertex(VertexProperties&&... properties)
225{
226 /* Get the vertex index. */
227 auto index = m_free_vertices.empty() ? m_next_vertex_index++ : m_free_vertices.back();
228
229 /* Create the vertex. */
230 m_vertices.emplace(index, V(index, std::forward<VertexProperties>(properties)...));
231
232 /* Initialize the data structures. */
233 if (m_free_vertices.empty())
234 {
235 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Forward> {}).emplace(index, EdgeIndexSet());
236 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Backward> {}).emplace(index, EdgeIndexSet());
237 boost::hana::at_key(m_degrees, boost::hana::type<Forward> {}).emplace(index, 0);
238 boost::hana::at_key(m_degrees, boost::hana::type<Backward> {}).emplace(index, 0);
239 }
240 else
241 {
242 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Forward> {}).at(index).clear();
243 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Backward> {}).at(index).clear();
244 boost::hana::at_key(m_degrees, boost::hana::type<Forward> {}).at(index) = 0;
245 boost::hana::at_key(m_degrees, boost::hana::type<Backward> {}).at(index) = 0;
246 }
247
248 if (!m_free_vertices.empty())
249 {
250 // If m_free_vertices was non-empty, we additionally need to pop_back the used index.
251 m_free_vertices.pop_back();
252 }
253
254 return index;
255}

257template<IsVertex V, IsEdge E>
258template<typename... EdgeProperties>
259 requires HasEdgeProperties<E, EdgeProperties...>
260EdgeIndex DynamicGraph<V, E>::add_directed_edge(VertexIndex source, VertexIndex target, EdgeProperties&&... properties)
261{
262 vertex_index_check(source, "DynamicGraph<V, E>::add_directed_edge(...): Source vertex does not exist.");
263 vertex_index_check(target, "DynamicGraph<V, E>::add_directed_edge(...): Target vertex does not exist.");
264
265 /* Get the edge index */
266 const auto index = m_free_edges.empty() ? m_next_edge_index++ : m_free_edges.back();
267
268 /* Create the edge */
269 m_edges.emplace(index, E(index, source, target, std::forward<EdgeProperties>(properties)...));
270
271 /* Initialize the data structures. */
272 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Forward> {}).at(source).insert(index);
273 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Backward> {}).at(target).insert(index);
274 ++boost::hana::at_key(m_degrees, boost::hana::type<Forward> {}).at(source);
275 ++boost::hana::at_key(m_degrees, boost::hana::type<Backward> {}).at(target);
276
277 if (!m_free_edges.empty())
278 {
279 // If m_free_edges was non-empty, we additionally need to pop_back the used index.
280 m_free_edges.pop_back();
281 }
282
283 return index;
284}

286template<IsVertex V, IsEdge E>
287template<typename... EdgeProperties>
288 requires HasEdgeProperties<E, EdgeProperties...>
289std::pair<EdgeIndex, EdgeIndex> DynamicGraph<V, E>::add_undirected_edge(VertexIndex source, VertexIndex target, EdgeProperties&&... properties)
290{
291 auto properties_tuple = std::make_tuple(std::forward<EdgeProperties>(properties)...);
292 auto properties_tuple_copy = properties_tuple;
293
294 const auto forward_edge_index =
295 std::apply([this, source, target](auto&&... args) { return this->add_directed_edge(source, target, std::forward<decltype(args)>(args)...); },
296 std::move(properties_tuple_copy));
297 const auto backward_edge_index =
298 std::apply([this, source, target](auto&&... args) { return this->add_directed_edge(target, source, std::forward<decltype(args)>(args)...); },
299 std::move(properties_tuple));
300
301 return std::make_pair(forward_edge_index, backward_edge_index);
302}

304template<IsVertex V, IsEdge E>
305void DynamicGraph<V, E>::remove_vertex(VertexIndex vertex)
306{
307 vertex_index_check(vertex, "DynamicGraph<V, E>::remove_vertex(...): Vertex does not exist.");
308
309 /* Remove backward adjacent edges from vertex of adjacent vertices */
310 for (const auto& edge : get_adjacent_edge_indices<Forward>(vertex))
311 {
312 const auto target = get_target<Forward>(edge);
313 if (target == vertex)
314 {
315 // Ignore loops over vertex.
316 continue;
317 }
318
319 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Backward> {}).at(target).erase(edge);
320 --boost::hana::at_key(m_degrees, boost::hana::type<Backward> {}).at(target);
321 m_edges.erase(edge);
322 m_free_edges.push_back(edge);
323 }
324 /* Remove forward adjacent edges to vertex of adjacent vertices */
325 for (const auto& edge : get_adjacent_edge_indices<Backward>(vertex))
326 {
327 const auto target = get_target<Backward>(edge);
328 if (target == vertex)
329 {
330 // Ignore loops over vertex.
331 continue;
332 }
333
334 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Forward> {}).at(target).erase(edge);
335 --boost::hana::at_key(m_degrees, boost::hana::type<Forward> {}).at(target);
336 m_edges.erase(edge);
337 m_free_edges.push_back(edge);
338 }
339
340 m_vertices.erase(vertex);
341 m_free_vertices.push_back(vertex);
342}

344template<IsVertex V, IsEdge E>
345void DynamicGraph<V, E>::remove_edge(EdgeIndex edge)
346{
347 edge_index_check(edge, "DynamicGraph<V, E>::remove_edge(...): Edge does not exist.");
348
349 const auto source = get_source<Forward>(edge);
350 const auto target = get_target<Forward>(edge);
351
352 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Forward> {}).at(source).erase(edge);
353 boost::hana::at_key(m_adjacent_edges, boost::hana::type<Backward> {}).at(target).erase(edge);
354 --boost::hana::at_key(m_degrees, boost::hana::type<Forward> {}).at(source);
355 --boost::hana::at_key(m_degrees, boost::hana::type<Backward> {}).at(target);
356 m_edges.erase(edge);
357 m_free_edges.push_back(edge);
358}
