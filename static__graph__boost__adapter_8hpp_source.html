226template<IsStaticGraph Graph, IsDirection Direction>
227std::pair<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type,
228 std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type>>
229strong_components(const DirectionTaggedType<Graph, Direction>& g)
230{
231 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
232 using vertices_size_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type;
233
234 auto vertex_index_map = TrivialReadPropertyMap<vertex_descriptor_type, vertex_descriptor_type>();
235 auto c = std::vector<vertices_size_type>(g.get().get_num_vertices());
236 auto component_map = VectorReadWritePropertyMap<vertex_descriptor_type, vertices_size_type>(c);
237
238 const auto num_components = boost::strong_components(g, component_map, boost::vertex_index_map(vertex_index_map));
239 return std::make_pair(num_components, c);
240}

242template<IsStaticGraph Graph, IsDirection Direction>
243IndexGroupedVector<std::pair<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type,
244 typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor>>
245get_partitioning(typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type num_components,
246 std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type> component_map)
247{
248 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
249 using vertices_size_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertices_size_type;
250 using state_component_pair_t = std::pair<vertices_size_type, vertex_descriptor_type>;
251
252 auto partitioning = std::vector<state_component_pair_t>();
253 for (vertex_descriptor_type v = 0; v < component_map.size(); ++v)
254 {
255 partitioning.push_back({ component_map.at(v), v });
256 }
257 std::sort(std::begin(partitioning), std::end(partitioning));
258 return IndexGroupedVector<state_component_pair_t>::create(std::move(partitioning),
259 [](const auto& prev, const auto& cur) { return prev.first != cur.first; });
260}

266template<IsStaticGraph Graph, IsDirection Direction, class SourceInputIter>
267std::tuple<std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor>, ContinuousCostList>
268dijkstra_shortest_paths(const DirectionTaggedType<Graph, Direction>& g, const ContinuousCostList& w, SourceInputIter s_begin, SourceInputIter s_end)
269{
270 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
271 using edge_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::edge_descriptor;
272
273 auto p = std::vector<vertex_descriptor_type>(g.get().get_num_vertices());
274 auto predecessor_map = VectorReadWritePropertyMap<vertex_descriptor_type, vertex_descriptor_type>(p);
275 auto d = ContinuousCostList(g.get().get_num_vertices());
276 auto distance_map = VectorReadWritePropertyMap<vertex_descriptor_type, ContinuousCost>(d);
277 auto weight_map = VectorReadPropertyMap<edge_descriptor_type, ContinuousCost>(w);
278 auto vertex_index_map = TrivialReadPropertyMap<vertex_descriptor_type, vertex_descriptor_type>();
279 auto compare = std::less<ContinuousCost>();
280 auto combine = std::plus<ContinuousCost>();
281 auto inf = std::numeric_limits<ContinuousCost>::infinity();
282 auto zero = ContinuousCost();
283
284 // multiple source shortest path version.
285 boost::dijkstra_shortest_paths(g, //
286 s_begin,
287 s_end,
288 predecessor_map,
289 distance_map,
290 weight_map,
291 vertex_index_map,
292 compare,
293 combine,
294 inf,
295 zero,
296 boost::default_dijkstra_visitor());
297
298 return std::make_tuple(p, d);
299};

305template<typename VertexDescriptor>
306struct BFSBoostVisitor : public boost::bfs_visitor<>
307{
308 std::reference_wrapper<std::vector<VertexDescriptor>> predecessors;
309 std::reference_wrapper<DiscreteCostList> distances;
310
311 BFSBoostVisitor(std::vector<VertexDescriptor>& p, DiscreteCostList& d) : predecessors(p), distances(d) {}
312
313 template<typename Edge, typename Graph>
314 void tree_edge(Edge e, const Graph& g) const
315 {
316 auto u = source(e, g);
317 auto v = target(e, g);
318 predecessors.get().at(v) = u;
319 distances.get().at(v) = distances.get().at(u) + 1;
320 }
321};

323template<IsStaticGraph Graph, IsDirection Direction, class SourceInputIter>
324std::tuple<std::vector<typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor>, DiscreteCostList>
325breadth_first_search(const DirectionTaggedType<Graph, Direction>& g, SourceInputIter s_begin, SourceInputIter s_end)
326{
327 using vertex_descriptor_type = typename boost::graph_traits<DirectionTaggedType<Graph, Direction>>::vertex_descriptor;
328 using ColorMap = boost::iterator_property_map<std::vector<boost::default_color_type>::iterator, boost::identity_property_map>;
329
330 auto buffer = boost::queue<vertex_descriptor_type>();
331 auto p = std::vector<vertex_descriptor_type>(g.get().get_num_vertices());
332 for (vertex_descriptor_type v = 0; v < g.get().get_num_vertices(); ++v)
333 {
334 p.at(v) = v;
335 }
336 auto inf = std::numeric_limits<DiscreteCost>::max();
337 auto d = DiscreteCostList(g.get().get_num_vertices(), inf);
338 for (auto it = s_begin; it != s_end; ++it)
339 {
340 d.at(*it) = DiscreteCost(0);
341 }
342 auto visitor = BFSBoostVisitor(p, d);
343 auto color_vector = std::vector<boost::default_color_type>(g.get().get_num_vertices(), boost::white_color);
344 auto color_map = ColorMap(color_vector.begin(), boost::identity_property_map());
345 boost::breadth_first_search(g, s_begin, s_end, buffer, visitor, color_map);
346
347 return std::make_tuple(p, d);
348}
