95template<formalism::IsFluentOrDerivedTag P>
96auto StateImpl::get_atoms() const
97{
98 if constexpr (std::is_same_v<P, formalism::FluentTag>)
99 {
100 return m_fluent_atoms->compressed_range();
101 }
102 else if constexpr (std::is_same_v<P, formalism::DerivedTag>)
103 {
104 return m_derived_atoms->compressed_range();
105 }
106 else
107 {
108 static_assert(dependent_false<P>::value, "Missing implementation for IsStaticOrFluentOrDerivedTag.");
109 }
110}

112template<formalism::IsFluentOrDerivedTag P, std::ranges::forward_range Range1, std::ranges::forward_range Range2>
113 requires IsRangeOver<Range1, Index> && IsRangeOver<Range2, Index>
114bool StateImpl::literals_hold(const Range1& positive_atoms, const Range2& negative_atoms) const
115{
116 return is_supseteq(get_atoms<P>(), positive_atoms) && are_disjoint(get_atoms<P>(), negative_atoms);
117}
