72 template<typename Graph>
73 requires IsVertexListGraph<Graph> && IsEdgeListGraph<Graph> && IsVertexColoredGraph<Graph>
74 explicit SparseGraph(const Graph& graph) : SparseGraph()
75 {
76 if (is_multi(graph))
77 {
78 throw std::runtime_error("SparseGraph::SparseGraph(graph): Multi graph is not supported.");
79 }
80 if (!is_loopless(graph))
81 {
82 throw std::runtime_error("SparseGraph::SparseGraph(graph): looping edges are not supported.");
83 }
84 if (!is_undirected(graph))
85 {
86 throw std::runtime_error("SparseGraph::SparseGraph(graph): directed graphs are not supported due to bad performance. Please translate it into an "
87 "undirected graph before using nauty.");
88 }
89
90 /* Remap indices to 0,1,... indexing schema. */
91 auto remap = mimir::IndexMap<uint32_t>();
92 for (const auto& vertex : graph.get_vertices())
93 {
94 remap.emplace(vertex.get_index(), remap.size());
95 }
96
97 // Copies from nauty sparsegraph and wrapped into c++ types.
98 size_t nde; /* Number of directed edges (loops contribute only 1) */
99 std::vector<size_t> v; /* Array of indexes into e[*] */
100 int nv; /* Number of vertices */
101 std::vector<int> d; /* Array with out-degree of each vertex */
102 std::vector<int> e; /* Array to hold lists of neighbours */
103 size_t vlen, dlen, elen; /* Sizes of arrays in units of type */
104
105 nv = graph.get_num_vertices(); /* Number of directed edges (loops contribute only 1) */
106 nde = graph.get_num_edges(); /* Number of vertices */
107 vlen = nv;
108 dlen = nv;
109 elen = nde;
110
111 d.reserve(nv);
112 v.reserve(nv);
113 e.reserve(nde);
114 for (const auto& vertex : graph.get_vertices())
115 {
116 d.push_back(graph.template get_degree<ForwardTag>(vertex.get_index()));
117 v.push_back(e.size());
118 for (const auto& edge : graph.template get_adjacent_edges<ForwardTag>(vertex.get_index()))
119 {
120 e.push_back(remap.at(edge.get_target()));
121 }
122 }
123
124 /* Add vertex coloring. */
125
126 auto color_vertex_pairs = std::vector<std::pair<Color, uint32_t>> {};
127 color_vertex_pairs.reserve(nv);
128 for (const auto& vertex : graph.get_vertices())
129 {
130 color_vertex_pairs.emplace_back(get_color(vertex), remap.at(vertex.get_index()));
131 }
132 std::sort(color_vertex_pairs.begin(), color_vertex_pairs.end());
133
134 auto coloring = ColorList {};
135 coloring.reserve(nv);
136 for (const auto& [color, _] : color_vertex_pairs)
137 {
138 coloring.push_back(color);
139 }
140
141 auto lab = std::vector<int>(nv, 0);
142 auto ptn = std::vector<int>(nv, 0);
143 for (int i = 1; i < nv; ++i)
144 {
145 lab[i - 1] = color_vertex_pairs[i - 1].second;
146 ptn[i - 1] = (color_vertex_pairs[i - 1].first != color_vertex_pairs[i].first) ? 0 : 1;
147 }
148 if (nv > 0)
149 {
150 lab[nv - 1] = color_vertex_pairs[nv - 1].second;
151 ptn[nv - 1] = 0;
152 }
153
154 initialize(nde, v, nv, d, e, vlen, dlen, elen, lab, ptn, coloring);
155 }
